22  public class SingleThreadedReadWrite {
23  
24      public static void main(final String[] args) throws NoSuchSessionException, TaskExecutionException,
25          ConfigurationException {
26          // init of test structures
27          int numBlocks = 50;
28          int address = 12345;
29          ByteBuffer writeData = ByteBuffer.allocate(512 * numBlocks);
30          ByteBuffer readData = ByteBuffer.allocate(512 * numBlocks);
31          Random random = new Random(System.currentTimeMillis());
32          random.nextBytes(writeData.array());
33  
34          // init of initiator and the session
35          String target = "testing-xen2-disk1";
36          Initiator initiator = new Initiator(Configuration.create());
37          initiator.createSession(target);
38  
39          // writing the data single threaded
40          initiator.write(target, writeData, address, writeData.capacity());
41  
42          // reading the data single threaded
43          initiator.read(target, readData, address, readData.capacity());
44  
45          // closing the session
46          initiator.closeSession(target);
47  
48          // correctness check
49          if (!Arrays.equals(writeData.array(), readData.array())) {
50              throw new IllegalStateException("Data read must be equal to the data written");
51          }
52      }
