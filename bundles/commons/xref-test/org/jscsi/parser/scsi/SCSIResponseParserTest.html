104     /**
105      * This test case validates the parsing process.
106      * 
107      * @throws IOException
108      *             This exception should be never thrown.
109      * @throws InternetSCSIException
110      *             This exception should be never thrown.
111      * @throws DigestException
112      *             This exception should be never thrown.
113      */
114     @Test
115     public void testDeserialize1() throws IOException, InternetSCSIException, DigestException {
116 
117         super.setUp(TEST_CASE_1);
118         super.testDeserialize(false, true, OperationCode.SCSI_RESPONSE, 0x00000000, 0x00000000, 0x34000000);
119 
120         assertTrue(recognizedParser instanceof SCSIResponseParser);
121 
122         SCSIResponseParser parser = (SCSIResponseParser)recognizedParser;
123 
124         // test SCSI response fields
125         assertFalse(parser.isBidirectionalReadResidualOverflow());
126         assertFalse(parser.isBidirectionalReadResidualUnderflow());
127         assertFalse(parser.isResidualOverflow());
128         assertFalse(parser.isResidualUnderflow());
129 
130         assertEquals(ServiceResponse.COMMAND_COMPLETED_AT_TARGET, parser.getResponse());
131         assertEquals(SCSIStatus.GOOD, parser.getStatus());
132         assertEquals(0x00000035, parser.getStatusSequenceNumber());
133         assertEquals(0x00000035, parser.getExpectedCommandSequenceNumber());
134         assertEquals(0x00000055, parser.getMaximumCommandSequenceNumber());
135         assertEquals(0x00000000, parser.getExpectedDataSequenceNumber());
136         assertEquals(0x00000000, parser.getBidirectionalReadResidualCount());
137         assertEquals(0x00000000, parser.getResidualCount());
138     }
