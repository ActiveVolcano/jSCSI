178     /**
179      * This test case validates the parsing process.
180      * 
181      * @throws IOException
182      *             This exception should be never thrown.
183      * @throws InternetSCSIException
184      *             This exception should be never thrown.
185      */
186     @Test
187     public void testDeserialize1() throws IOException, InternetSCSIException, DigestException {
188 
189         SettingsMap expectedKeyValuePair = new SettingsMap();
190         expectedKeyValuePair.add(OperationalTextKey.INITIATOR_NAME, "iqn.com.ibm.k-machine");
191         expectedKeyValuePair.add(OperationalTextKey.TARGET_NAME,
192             "iqn.1987-05.com.cisco.00.58031e1d068ac226d385847592c0b670.IBM-Disk");
193         expectedKeyValuePair.add(OperationalTextKey.AUTH_METHOD, "none");
194         expectedKeyValuePair.add(OperationalTextKey.SESSION_TYPE, "normal");
195 
196         super.setUp(TEST_CASE_1);
197         super.testDeserialize(true, false, OperationCode.LOGIN_REQUEST, 0x00000000, 0x00000095, 0xC8D04B81);
198         super.testDataSegment(expectedKeyValuePair);
199 
200         assertTrue(recognizedParser instanceof LoginRequestParser);
201 
202         LoginRequestParser parser = (LoginRequestParser)recognizedParser;
203 
204         // test login request fields
205         assertFalse(parser.isContinueFlag());
206         assertEquals(LoginStage.SECURITY_NEGOTIATION, parser.getCurrentStageNumber());
207         assertEquals(LoginStage.SECURITY_NEGOTIATION, parser.getNextStageNumber());
208         assertEquals((byte)0x00, parser.getMinVersion());
209         assertEquals((byte)0x00, parser.getMaxVersion());
210 
211         // test ISID fields
212         ISID expectedISID = new ISID(Format.OUI_FORMAT, (byte)0x00, (short)0x0000, (byte)0x00, (short)0xABCD);
213         assertTrue(expectedISID.equals(parser.getInitiatorSessionID()));
214 
215         assertEquals((short)0x0000, parser.getTargetSessionIdentifyingHandle());
216         assertEquals(0x00000000, parser.getConnectionID());
217         assertEquals(0x00000001, parser.getCommandSequenceNumber());
218         assertEquals(0x00000000, parser.getExpectedStatusSequenceNumber());
219     }
