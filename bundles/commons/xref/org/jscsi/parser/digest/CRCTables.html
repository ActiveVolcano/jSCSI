29  /**
30   * <h1>CRCTables</h1>
31   * <p>
32   * This class generates tables for CRC algorithm. It is possible to change the
33   * generator polynom and the offset. Usage:
34   * <ol>
35   * <li>Make new <code>CRCTables</code> object.</li>
36   * <li>Put <code>generatorPolynom</code> in the constructor.</li>
37   * <li>Invoke the method <code>getTable</code> with offset as an argument. The
38   * offset is an integer number, which specifies the number of bits to shift,
39   * e.g. for <code>A</code> offset is <code>56</code>, for <code>D</code>
40   * <code>32</code>.</li>
41   * <li>You will have CRC table(<code>0-255</code>) for the requested polynom.</li>
42   * </ol>
43   * 
44   * @author Stepan
45   */
46  public final class CRCTables {
47  
48      // --------------------------------------------------------------------------
49      // --------------------------------------------------------------------------
50  
51      /** The size of a table. */
52      private static final int SIZE_OF_TABLE = 256;
53  
54      /** Bit mask for throwing away the 33-rd bit. */
55      private static final long BIN_MASK_33BIT = 0x00000000FFFFFFFFL;
56  
57      /** Bit mask with 1 only in the first position. */
58      private static final long BIN_MASK_1FIRST_ONLY = 0x8000000000000000L;
59  
60      /** Bit mask with 1 only in the first position for int. */
61      private static final int BIN_MASK_1FIRST_ONLY_INT = 0x80000000;
62  
63      // --------------------------------------------------------------------------
64      // --------------------------------------------------------------------------
65  
66      /** The generator polynom. */
67      private final long generatorPolynom;
68  
69      // --------------------------------------------------------------------------
70      // --------------------------------------------------------------------------
71  
72      /**
73       * Constructor to create a new, empty <code>CRCTables</code> object with the
74       * initialized generator polynom.
75       * 
76       * @param initGeneratorPolynom
77       *            The generator polynom to use.
78       */
79      public CRCTables(final long initGeneratorPolynom) {
80  
81          // FIXME: Really needed?
82          generatorPolynom = initGeneratorPolynom;
83      }
84  
85      // --------------------------------------------------------------------------
86      // --------------------------------------------------------------------------
87  
88      /**
89       * Returns all remainders of the polynomial division for the given offset.
90       * 
91       * @param offset
92       *            The numbers of bits to shift, e.g. for <code>A</code> offset
93       *            is <code>56</code>, for <code>D</code> <code>32</code>.
94       * @return The CRC table with all remainders for the given offset.
95       */
96      public final int[] getTable(final int offset) {
97  
98          final int[] table = new int[SIZE_OF_TABLE];
99          long numberToCalculate;
100 
101         for (int number = 0; number < table.length; number++) {
102             numberToCalculate = Integer.reverseBytes(Integer.reverse(number));
103             table[number] = calculateCRC32(numberToCalculate << offset);
104         }
105 
106         return table;
107     }
108 
109     // --------------------------------------------------------------------------
110     // --------------------------------------------------------------------------
111 
112     /**
113      * Method for CRC calculation for the just one value (type long) Algorithm:
114      * Load the register with zero bits. Reverse message Augment the message by
115      * appending W zero bits to the end of it. While (more message bits) Begin
116      * Shift the register left by one bit, reading the next bit of the augmented
117      * message into register bit position 0. If (a 1 bit popped out of the
118      * register during step 3) Register = Register XOR Poly. End Reverse
119      * register The register now contains the CRC. Notes: W=32, that's why we
120      * have offsets 32, 40, 48 and 56 instead of 0,8,16,24 Size of register=W;
121      * 
122      * @param value
123      *            A <code>long</code> value, for which the CRC should be
124      *            calculated.
125      * @return The remainder of the polynomial division --> CRC.
126      */
127     protected final int calculateCRC32(final long value) {
128 
129         long l = value;
130         int register = 0;
131         // We are going to throw the first bit away. Then making int from long.
132         final int gxInt = (int) (generatorPolynom & BIN_MASK_33BIT);
133 
134         // the first bit before bit shifting in the input polynom
135         int bit = 0;
136 
137         for (int i = 0; i < Long.SIZE; i++) {
138             // is the highest bit set?
139             if ((l & BIN_MASK_1FIRST_ONLY) == 0) {
140                 bit = 0;
141             } else {
142                 bit = 1;
143             }
144 
145             // is the highest bit set?
146             if ((register & BIN_MASK_1FIRST_ONLY_INT) == 0) {
147                 register = register << 1;
148                 l = l << 1;
149                 register += bit;
150             } else {
151                 register = register << 1;
152                 l = l << 1;
153                 register += bit;
154                 register = register ^ gxInt;
155             }
156 
157         }
158 
159         return Integer.reverse(register);
160     }
