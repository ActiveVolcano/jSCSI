40  /**
41   * <h1>TaskManagementFunctionRequestParser</h1>
42   * <p>
43   * This class parses a Task Management Function Request message defined in the iSCSI Standard (RFC3720).
44   * <p>
45   * <h4>TotalAHSLength and DataSegmentLength</h4> For this PDU TotalAHSLength and DataSegmentLength MUST be
46   * <code>0</code>.
47   * <p>
48   * <h4>LUN</h4> This field is required for functions that address a specific LU (ABORT TASK, CLEAR TASK SET,
49   * ABORT TASK SET, CLEAR ACA, LOGICAL UNIT RESET) and is reserved in all others.
50   * <p>
51   * 
52   * @author Volker Wildi
53   */
54  public final class TaskManagementFunctionRequestParser extends InitiatorMessageParser {

59      /**
60       * This enumeration defines all valid function codes, which are defined in
61       * the iSCSI standard (RFC3720).
62       */
63      public static enum FunctionCode {
64          /**
65           * This is an artificial funcition code, which only used to indicate a
66           * not set function code.
67           */
68          UNSET((byte)0),
69          /** Aborts the task identified by the Referenced Task Tag field. */
70          ABORT_TASK((byte)1),
71  
72          /** Aborts all Tasks issued via this session on the logical unit. */
73          ABORT_TASK_SET((byte)2),
74  
75          /** Clears the Auto Contingent Allegiance condition. */
76          CLEAR_ACA((byte)3);
77  
78          private byte value;
79  
80          private static Map<Byte, FunctionCode> mapping;
81  
82          static {
83              FunctionCode.mapping = new HashMap<Byte, FunctionCode>();
84              for (FunctionCode s : values()) {
85                  FunctionCode.mapping.put(s.value, s);
86              }
87          }
88  
89          private FunctionCode(final byte newValue) {
90  
91              value = newValue;
92          }
93  
94          /**
95           * Returns the value of this enumeration.
96           * 
97           * @return The value of this enumeration.
98           */
99          public final byte value() {
100 
101             return value;
102         }
103 
104         /**
105          * Returns the constant defined for the given <code>value</code>.
106          * 
107          * @param value
108          *            The value to search for.
109          * @return The constant defined for the given <code>value</code>. Or <code>null</code>, if this value
110          *         is not defined by this
111          *         enumeration.
112          */
113         public static final FunctionCode valueOf(final byte value) {
114 
115             return FunctionCode.mapping.get(value);
116         }
117     }

190     /**
191      * For recovery purposes, the iSCSI target and initiator maintain a data
192      * acknowledgement reference number - the first input DataSN number
193      * unacknowledged by the initiator. When issuing a new command, this number
194      * is set to <code>0</code>. If the function is TASK REASSIGN, which
195      * establishes a new connection allegiance for a previously issued Read or
196      * Bidirectional command, <code>ExpDataSN</code> will contain an updated
197      * data acknowledgement reference number or the value <code>0</code>; the
198      * latter indicating that the data acknowledgement reference number is
199      * unchanged. The initiator MUST discard any data PDUs from the previous
200      * execution that it did not acknowledge and the target MUST transmit all
201      * Data-In PDUs (if any) starting with the data acknowledgement reference
202      * number. The number of retransmitted PDUs may or may not be the same as
203      * the original transmission depending on if there was a change in
204      * MaxRecvDataSegmentLength in the reassignment. The target MAY also send no
205      * more Data-In PDUs if all data has been acknowledged.
206      * <p/>
207      * The value of <code>ExpDataSN</code> MUST be <code>0</code> or higher than the <code>DataSN</code> of
208      * the last acknowledged Data-In PDU, but not larger than <code>DataSN+1</code> of the last Data-In PDU
209      * sent by the target. Any other value MUST be ignored by the target.
210      * <p/>
211      * For other functions this field is reserved.
212      * <p/>
213      * 
214      * @return The expected data sequence number of this <code>TaskManagementFunctionRequestParser</code>
215      *         object.
216      */
217     public final int getExpDataSN() {
218 
219         return expDataSN;
220     }

222     /**
223      * The Task Management functions provide an initiator with a way to
224      * explicitly control the execution of one or more Tasks (SCSI and iSCSI
225      * tasks). The Task Management function codes are listed below. For a more
226      * detailed description of SCSI task management, see [SAM2].
227      * <p>
228      * <table border="1">
229      * <tr>
230      * <th>Function Code</th>
231      * <th>Description</th>
232      * </tr>
233      * <tr>
234      * <td>1</td>
235      * <td>ABORT TASK - aborts the task identified by the Referenced Task Tag field.</td>
236      * </tr>
237      * <tr>
238      * <td>2</td>
239      * <td>ABORT TASK SET - aborts all Tasks issued via this session on the logical unit.</td>
240      * </tr>
241      * <tr>
242      * <td>3</td>
243      * <td>CLEAR ACA - clears the Auto Contingent Allegiance condition.</td>
244      * </tr>
245      * <tr>
246      * <td>4</td>
247      * <td>CLEAR TASK SET - aborts all Tasks in the appropriate task set as defined by the TST field in the
248      * Control mode page (see [SPC3]).</td>
249      * </tr>
250      * <tr>
251      * <td>5</td>
252      * <td>LOGICAL UNIT RESET</td>
253      * </tr>
254      * <tr>
255      * <td>6</td>
256      * <td>TARGET WARM RESET</td>
257      * </tr>
258      * <tr>
259      * <td>7</td>
260      * <td>TARGET COLD RESET</td>
261      * </tr>
262      * <tr>
263      * <td>8</td>
264      * <td>TASK REASSIGN - reassigns connection allegiance for the task identified by the Referenced Task Tag
265      * field to this connection, thus resuming the iSCSI exchanges for the task.</td>
266      * </tr>
267      * </table>
268      * <p>
269      * For all these functions, the Task Management function response MUST be returned as detailed in Section
270      * 10.6 Task Management Function Response. All these functions apply to the referenced tasks regardless of
271      * whether they are proper SCSI tasks or tagged iSCSI operations. Task management requests must act on all
272      * the commands from the same session having a CmdSN lower than the task management CmdSN. LOGICAL UNIT
273      * RESET, TARGET WARM RESET and TARGET COLD RESET may affect commands from other sessions or commands from
274      * the same session with CmdSN equal or exceeding CmdSN.
275      * <p>
276      * If the task management request is marked for immediate delivery, it must be considered immediately for
277      * execution, but the operations involved (all or part of them) may be postponed to allow the target to
278      * receive all relevant tasks. According to [SAM2], for all the tasks covered by the Task Management
279      * response (i.e., with CmdSN lower than the task management command CmdSN) but except the Task Management
280      * response to a TASK REASSIGN, additional responses MUST NOT be delivered to the SCSI layer after the
281      * Task Management response. The iSCSI initiator MAY deliver to the SCSI layer all responses received
282      * before the Task Management response (i.e., it is a matter of implementation if the SCSI responses,
283      * received before the Task Management response but after the task management request was issued, are
284      * delivered to the SCSI layer by the iSCSI layer in the initiator). The iSCSI target MUST ensure that no
285      * responses for the tasks covered by a task management function are delivered to the iSCSI initiator
286      * after the Task Management response except for a task covered by a TASK REASSIGN.
287      * <p>
288      * For ABORT TASK SET and CLEAR TASK SET, the issuing initiator MUST continue to respond to all valid
289      * target transfer tags (received via R2T, Text Response, NOP-In, or SCSI Data-In PDUs) related to the
290      * affected task set, even after issuing the task management request. The issuing initiator SHOULD however
291      * terminate (i.e., by setting the F-bit to 1) these response sequences as quickly as possible. The target
292      * on its part MUST wait for responses on all affected target transfer tags before acting on either of
293      * these two task management requests. In case all or part of the response sequence is not received (due
294      * to digest errors) for a valid TTT, the target MAY treat it as a case of within-command error recovery
295      * class (see Section 6.1.4.1 Recovery Within-command) if it is supporting ErrorRecoveryLevel >= 1, or
296      * alternatively may drop the connection to complete the requested task set function.
297      * <p>
298      * If an ABORT TASK is issued for a task created by an immediate command then RefCmdSN MUST be that of the
299      * Task Management request itself (i.e., CmdSN and RefCmdSN are equal); otherwise RefCmdSN MUST be set to
300      * the CmdSN of the task to be aborted (lower than CmdSN).
301      * <p>
302      * If the connection is still active (it is not undergoing an implicit or explicit logout), ABORT TASK
303      * MUST be issued on the same connection to which the task to be aborted is allegiant at the time the Task
304      * Management Request is issued. If the connection is implicitly or explicitly logged out (i.e., no other
305      * request will be issued on the failing connection and no other response will be received on the failing
306      * connection), then an ABORT TASK function request may be issued on another connection. This Task
307      * Management request will then establish a new allegiance for the command to be aborted as well as abort
308      * it (i.e., the task to be aborted will not have to be retried or reassigned, and its status, if issued
309      * but not acknowledged, will be reissued followed by the Task Management response).
310      * <p>
311      * At the target an ABORT TASK function MUST NOT be executed on a Task Management request; such a request
312      * MUST result in Task Management response of "Function rejected".
313      * <p>
314      * For the LOGICAL UNIT RESET function, the target MUST behave as dictated by the Logical Unit Reset
315      * function in [SAM2].
316      * <p>
317      * The implementation of the TARGET WARM RESET function and the TARGET COLD RESET function is OPTIONAL and
318      * when implemented, should act as described below. The TARGET WARM RESET is also subject to SCSI access
319      * controls on the requesting initiator as defined in [SPC3]. When authorization fails at the target, the
320      * appropriate response as described in Section 10.6 Task Management Function Response MUST be returned by
321      * the target. The TARGET COLD RESET function is not subject to SCSI access controls, but its execution
322      * privileges may be managed by iSCSI mechanisms such as login authentication.
323      * <p>
324      * When executing the TARGET WARM RESET and TARGET COLD RESET functions, the target cancels all pending
325      * operations on all Logical Units known by the issuing initiator. Both functions are equivalent to the
326      * Target Reset function specified by [SAM2]. They can affect many other initiators logged in with the
327      * servicing SCSI target port.
328      * <p>
329      * The target MUST treat the TARGET COLD RESET function additionally as a power on event, thus terminating
330      * all of its TCP connections to all initiators (all sessions are terminated). For this reason, the
331      * Service Response (defined by [SAM2]) for this SCSI task management function may not be reliably
332      * delivered to the issuing initiator port. For the TASK REASSIGN function, the target should reassign the
333      * connection allegiance to this new connection (and thus resume iSCSI exchanges for the task). TASK
334      * REASSIGN MUST ONLY be received by the target after the connection on which the command was previously
335      * executing has been successfully logged-out. The Task Management response MUST be issued before the
336      * reassignment becomes effective. For additional usage semantics see Section 6.2 Retry and Reassign in
337      * Recovery.
338      * <p>
339      * At the target a TASK REASSIGN function request MUST NOT be executed to reassign the connection
340      * allegiance of a Task Management function request, an active text negotiation task, or a Logout task;
341      * such a request MUST result in Task Management response of "Function rejected".
342      * <p>
343      * TASK REASSIGN MUST be issued as an immediate command.
344      * 
345      * @return The function code of this <code>TaskManagementFunctionRequestParser</code> object.
346      */
347     public final FunctionCode getFunction() {
348 
349         return functionCode;

419     /** {@inheritDoc} */
420     @Override
421     protected final void checkIntegrity() throws InternetSCSIException {
422 
423         String exceptionMessage;
424 
425         do {
426             BasicHeaderSegment bhs = protocolDataUnit.getBasicHeaderSegment();
427             if (bhs.getTotalAHSLength() != 0) {
428                 exceptionMessage = "TotalAHSLength must be 0!";
429                 break;
430             }
431 
432             if (bhs.getDataSegmentLength() != 0) {
433                 exceptionMessage = "DataSegmentLength must be 0!";
434                 break;
435             }
436 
437             if (functionCode != FunctionCode.ABORT_TASK && functionCode != FunctionCode.ABORT_TASK_SET
438                 && functionCode != FunctionCode.CLEAR_ACA) {
439                 Utils.isReserved(logicalUnitNumber);
440             }
441 
442             // message is checked correctly
443             return;
444         } while (false);
445 
446         throw new InternetSCSIException(exceptionMessage);
447     }
