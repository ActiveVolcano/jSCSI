115 public enum SCSIStatus {
116     /**
117      * This status indicates that the device server has successfully completed
118      * the task.
119      */
120     GOOD((byte) 0x00),
121     /**
122      * This status indicates that an CA or ACA condition has occurred (see
123      * 5.9.1). Autosense data may be delivered (see 5.9.4.3)[SAM2].
124      */
125     CHECK_CONDITION((byte) 0x02),
126     /**
127      * This status shall be returned whenever the requested operation specified
128      * by an unlinked command is satisfied (see the PRE-FETCH commands in the
129      * SBC standard).
130      */
131     CONDITION_MET((byte) 0x04),
132     /**
133      * This status indicates that the logical unit is busy. This status shall be
134      * returned whenever a logical unit is temporarily unable to accept a
135      * command. The recommended application client recovery action is to issue
136      * the command again at a later time. If the UA_INTLCK_CTRL field in the
137      * Control mode page contains 11b (see SPC-3), termination of a command with
138      * BUSY status shall cause an unit attention condition to be established for
139      * the SCSI initiator port that sent the command with an additional sense
140      * code of PREVIOUS BUSY STATUS unless a PREVIOUS BUSY STATUS unit attention
141      * condition already exists.
142      */
143     BUSY((byte) 0x08),
144     /**
145      * This status or INTERMEDIATE-CONDITION MET shall be returned for each
146      * successfully completed command in a series of linked commands (except the
147      * last command), unless the command is termi- nated with CHECK CONDITION,
148      * RESERVATION CONFLICT, TASK SET FULL, or BUSY status. If INTERME- DIATE or
149      * INTERMEDIATE-CONDITION MET status is not returned, the series of linked
150      * commands is terminated and the task is ended. This status is the
151      * equivalent of GOOD status for linked commands.
152      */
153     INTERMEDIATE((byte) 0x10),
154     /**
155      * This status is returned whenever the requested operation specified by a
156      * linked command is satisfied (see the PRE-FETCH commands in the SBC
157      * standard), unless the command is termi- nated with CHECK CONDITION,
158      * RESERVATION CONFLICT, TASK SET FULL, or BUSY status. If INTERME- DIATE or
159      * INTERMEDIATE-CONDITION MET status is not returned, the series of linked
160      * commands is terminated and the task is ended.
161      */
162     INTERMEDIATE_CONDITION_MET((byte) 0x14),
163     /**
164      * This status shall be returned whenever a SCSI initiator port attempts to
165      * access a logical unit or an element of a logical unit in a way that
166      * conflicts with an existing reservation. (See the RESERVE, RELEASE,
167      * PERSISTENT RESERVE OUT and PERSISTENT RESERVE IN commands in SPC-2).
168      * <p>
169      * If the UA_INTLCK_CTRL field in the Control mode page contains 11b (see
170      * SPC-3), termination of a command with RESERVATION CONFLICT status shall
171      * cause an unit attention condition to be established for the SCSI
172      * initiator port that sent the command with an additional sense code of
173      * PREVIOUS RESERVATION CONFLICT STATUS unless a PREVIOUS RESERVATION
174      * CONFLICT STATUS unit attention condition already exists.
175      */
176     RESERVATION_CONFLICT((byte) 0x18),
177     /**
178      * This status shall be implemented if the logical unit supports the
179      * creation of tagged tasks (see 4.10)[SAM2]. This status shall not be
180      * implemented if the logical unit does not support the creation of tagged
181      * tasks.
182      * <p>
183      * When the logical unit has at least one task in the task set for a SCSI
184      * initiator port and a lack of task set resources prevents accepting a
185      * received tagged task from that SCSI initiator port into the task set,
186      * TASK SET FULL shall be returned. When the logical unit has no task in the
187      * task set for a SCSI initiator port and a lack of task set resources
188      * prevents accepting a received tagged task from that SCSI initiator port
189      * into the task set, BUSY should be returned.
190      * <p>
191      * When the logical unit has at least one task in the task set and a lack of
192      * task set resources prevents accepting a received untagged task into the
193      * task set, BUSY should be returned.
194      * <p>
195      * The logical unit should allow at least one command in the task set for
196      * each supported SCSI initiator port that has identified itself to the SCSI
197      * target port by a SCSI transport protocol specific procedure or by the
198      * successful trans- mission of a command.
199      * <p>
200      * If the UA_INTLCK_CTRL field in the Control mode page contains 11b (see
201      * SPC-3), termination of a command with TASK SET FULL status shall cause an
202      * unit attention condition to be established for the SCSI initiator port
203      * that sent the command with an additional sense code of PREVIOUS TASK SET
204      * FULL STATUS unless a PREVIOUS TASK SET FULL STATUS unit attention
205      * condition already exists.
206      */
207     TASK_SET_FULL((byte) 0x28),
208     /**
209      * This status shall be returned when an ACA exists within a task set and a
210      * SCSI initiator port issues a command for that task set when at least one
211      * of the following is true:
212      * <ol type="a">
213      * <li>There is a task with the ACA attribute (see 7.5.4)[SAM2] in the task
214      * set;</li>
215      * <li>The SCSI initiator port issuing the command did not cause the ACA
216      * condition; or</li>
217      * <li>The task created to process the command did not have the ACA
218      * attribute and the NACA bit was set to one in the CDB CONTROL byte of the
219      * faulting command (see 5.9.1)[SAM2].</li>
220      * </ol>
221      * The SCSI initiator port may reissue the command after the ACA condition
222      * has been cleared.
223      */
224     ACA_ACTIVE((byte) 0x30),
225     /**
226      * This status shall be returned when a task is aborted by another SCSI
227      * initiator port and the Control mode page TAS bit is set to one (see
228      * 5.7.3)[SAM2].
229      */
230     TASK_ABORTED((byte) 0x40);
