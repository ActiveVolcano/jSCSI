40  /**
41   * <h1>SCSICommandParser</h1>
42   * <p>
43   * This class parses a SCSI Command message defined in the iSCSI Standard
44   * (RFC3720).
45   * <p>
46   * <h4>CmdSN - Command Sequence Number</h4> Enables ordered delivery across
47   * multiple connections in a single session.
48   * <p>
49   * <h4>ExpStatSN</h4> Command responses up to
50   * <code>ExpStatSN - 1 (mod 2**32)</code> have been received (acknowledges
51   * status) on the connection.
52   * <p>
53   * <h4>Data Segment - Command Data</h4> Some SCSI commands require additional
54   * parameter data to accompany the SCSI command. This data may be placed beyond
55   * the boundary of the iSCSI header in a data segment. Alternatively, user data
56   * (e.g., from a WRITE operation) can be placed in the data segment (both cases
57   * are referred to as immediate data). These data are governed by the rules for
58   * solicited vs. unsolicited data outlined in Section 3.2.4.2 Data Transfer
59   * Overview.
60   * 
61   * @author Volker Wildi
62   */
63  public class SCSICommandParser extends InitiatorMessageParser {

68      /**
69       * This enumeration defines all valid types of additional header segments,
70       * which are defined by the iSCSI standard (RFC3720).
71       * <p>
72       * <table border="1">
73       * <tr>
74       * <th>Value</th>
75       * <th>Meaning</th>
76       * </tr>
77       * <tr>
78       * <td>0</td>
79       * <td>Reserved</td>
80       * </tr>
81       * <tr>
82       * <td>1</td>
83       * <td>Extended CDB</td>
84       * </tr>
85       * <tr>
86       * <td>2</td>
87       * <td>Expected Bidirectional Read Data Length</td>
88       * </tr>
89       * <tr>
90       * <td>3 - 63</td>
91       * <td>Reserved</td>
92       * </tr>
93       * </table>
94       */
95      public enum TaskAttributes {
96  
97          /**
98           * Untagged queuing allows a target to accept a command from an
99           * initiator for a logical unit or target routine while I/O processes
100          * from other initiators are being executed. Only one command for each
101          * I_T_x nexus shall be accepted at a time.
102          */
103         UNTAGGED((byte) 0),
104         /**
105          * If only SIMPLE QUEUE TAG messages are used, the target may execute
106          * the commands in any order that is deemed desirable within the
107          * constraints of the queue management algorithm specified in the
108          * control mode page (see [SAM2, 8.3.3.1]).
109          */
110         SIMPLE((byte) 1),
111         /**
112          * If ORDERED QUEUE TAG messages are used, the target shall execute the
113          * commands in the order received with respect to other commands
114          * received with ORDERED QUEUE TAG messages. All commands received with
115          * a SIMPLE QUEUE TAG message prior to a command received with an
116          * ORDERED QUEUE TAG message, regardless of initiator, shall be executed
117          * before that command with the ORDERED QUEUE TAG message. All commands
118          * received with a SIMPLE QUEUE TAG message after a command received
119          * with an ORDERED QUEUE TAG message, regardless of initiator, shall be
120          * executed after that command with the ORDERED QUEUE TAG message.
121          */
122         ORDERED((byte) 2),
123         /**
124          * A command received with a HEAD OF QUEUE TAG message is placed first
125          * in the queue, to be executed next. A command received with a HEAD OF
126          * QUEUE TAG message shall be executed prior to any queued I/O process.
127          * Consecutive commands received with HEAD OF QUEUE TAG messages are
128          * executed in a last-in-first-out order.
129          */
130         HEAD_OF_QUEUE((byte) 3),
131         /** ACA is Auto Contingent Allegiance. */
132         ACA((byte) 4);
133 
134         private final byte value;
135 
136         private static Map<Byte, TaskAttributes> mapping;
137 
138         static {
139             TaskAttributes.mapping = new HashMap<Byte, TaskAttributes>();
140             for (TaskAttributes s : values()) {
141                 TaskAttributes.mapping.put(s.value, s);
142             }
143         }
144 
145         private TaskAttributes(final byte newValue) {
146 
147             value = newValue;
148         }

291     /**
292      * For unidirectional operations, the Expected Data Transfer Length field
293      * contains the number of bytes of data involved in this SCSI operation. For
294      * a unidirectional write operation (W flag set to <code>1</code> and R flag
295      * set to <code>0</code>), the initiator uses this field to specify the
296      * number of bytes of data it expects to transfer for this operation. For a
297      * unidirectional read operation (W flag set to <code>0</code> and R flag
298      * set to <code>1</code>), the initiator uses this field to specify the
299      * number of bytes of data it expects the target to transfer to the
300      * initiator. It corresponds to the SAM2 byte count.
301      * <p>
302      * For bidirectional operations (both R and W flags are set to
303      * <code>1</code> ), this field contains the number of data bytes involved
304      * in the write transfer. For bidirectional operations, an additional header
305      * segment MUST be present in the header sequence that indicates the
306      * Bidirectional Read Expected Data Transfer Length. The Expected Data
307      * Transfer Length field and the Bidirectional Read Expected Data Transfer
308      * Length field correspond to the SAM2 byte count.
309      * <p>
310      * If the Expected Data Transfer Length for a write and the length of the
311      * immediate data part that follows the command (if any) are the same, then
312      * no more data PDUs are expected to follow. In this case, the F bit MUST be
313      * set to <code>1</code>.
314      * <p>
315      * If the Expected Data Transfer Length is higher than the FirstBurstLength
316      * (the negotiated maximum amount of unsolicited data the target will
317      * accept), the initiator MUST send the maximum amount of unsolicited data
318      * OR ONLY the immediate data, if any.
319      * <p>
320      * Upon completion of a data transfer, the target informs the initiator
321      * (through residual counts) of how many bytes were actually processed (sent
322      * and/or received) by the target.
323      * 
324      * @return The Expected Data Transfer Length of this SCSICommandParser
325      *         object.
326      */

328

417     /** {@inheritDoc} */
418     @Override
419     protected final void deserializeBytes1to3(final int line)
420             throws InternetSCSIException {
421 
422         readExpectedFlag = Utils.isBitSet(line & READ_EXPECTED_FLAG_MASK);
423         writeExpectedFlag = Utils.isBitSet(line & WRITE_EXPECTED_FLAG_MASK);
424         taskAttributes = TaskAttributes
425                 .valueOf((byte) ((line & TASK_ATTRIBUTES_FLAG_MASK) >> Constants.TWO_BYTES_SHIFT));
426 
427         Utils.isReserved(line & Constants.LAST_TWO_BYTES_MASK);
428     }

474     /** {@inheritDoc} */
475     @Override
476     protected final void checkIntegrity() throws InternetSCSIException {
477 
478         String exceptionMessage;
479         do {
480             if (!writeExpectedFlag
481                     && !protocolDataUnit.getBasicHeaderSegment().isFinalFlag()) {
482                 exceptionMessage = "W and F flag cannot both be 0.";
483                 break;
484             }
485 
486             if (expectedDataTransferLength != 0
487                     && !(readExpectedFlag || writeExpectedFlag)) {
488                 exceptionMessage = "The ExpectedDataTransferLength is greater than 0, so Read or/and Write Flag has to be set.";
489                 break;
490             }
491 
492             // message is checked correctly
493             return;
494 
495         } while (false);
496 
497         throw new InternetSCSIException(exceptionMessage);
498     }

503     /** {@inheritDoc} */
504     @Override
505     protected final int serializeBytes1to3() {
506 
507         int line = 0;
508 
509         line |= taskAttributes.value() << Constants.TWO_BYTES_SHIFT;
510 
511         if (writeExpectedFlag) {
512             line |= WRITE_EXPECTED_FLAG_MASK;
513         }
514 
515         if (readExpectedFlag) {
516             line |= READ_EXPECTED_FLAG_MASK;
517         }
518 
519         return line;
520     }
