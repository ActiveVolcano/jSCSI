51      /**
52       * Enumeration of all valid asynchronous message event codes.
53       * <table * border="1">
54       * <tr>
55       * <th>Event Code</th>
56       * <th>Description</th>
57       * </tr>
58       * <tr>
59       * <td>0</td>
60       * <td>A SCSI Asynchronous Event is reported in the sense data. Sense Data
61       * that accompanies the report, in the data segment, identifies the
62       * condition. The sending of a SCSI Event (Asynchronous Event Reporting in
63       * SCSI terminology) is dependent on the target support for SCSI
64       * asynchronous event reporting (see [SAM2]) as indicated in the standard
65       * INQUIRY data (see [SPC3]). Its use may be enabled by parameters in the
66       * SCSI Control mode page (see [SPC3]).</td>
67       * </tr>
68       * <tr>
69       * <td>1</td>
70       * <td>Target requests Logout. This Async Message MUST be sent on the same
71       * connection as the one requesting to be logged out. The initiator MUST
72       * honor this request by issuing a Logout as early as possible, but no later
73       * than Parameter3 seconds. Initiator MUST send a Logout with a reason code
74       * of "Close the connection" OR "Close the session" to close all the
75       * connections. Once this message is received, the initiator SHOULD NOT
76       * issue new iSCSI commands on the connection to be logged out. The target
77       * MAY reject any new I/O requests that it receives after this Message with
78       * the reason code "Waiting for Logout". If the initiator does not Logout in
79       * Parameter3 seconds, the target should send an Async PDU with iSCSI event
80       * code "Dropped the connection" if possible, or simply terminate the
81       * transport connection. Parameter1 and Parameter2 are reserved.</td>
82       * </tr>
83       * <tr>
84       * <td>2</td>
85       * <td>Target indicates it will drop the connection. The Parameter1 field
86       * indicates the CID of the connection that is going to be dropped. The
87       * Parameter2 field (Time2Wait) indicates, in seconds, the minimum time to
88       * wait before attempting to reconnect or reassign. The Parameter3 field
89       * (Time2Retain) indicates the maximum time allowed to reassign commands
90       * after the initial wait (in Parameter2). If the initiator does not attempt
91       * to reconnect and/or reassign the outstanding commands within the time
92       * specified by Parameter3, or if Parameter3 is 0, the target will terminate
93       * all outstanding commands on this connection. In this case, no other
94       * responses should be expected from the target for the outstanding commands
95       * on this connection. A value of 0 for Parameter2 indicates that reconnect
96       * can be attempted immediately.</td>
97       * </tr>
98       * <tr>
99       * <td>3</td>
100      * <td>Target indicates it will drop all the connections of this session.
101      * Parameter1 field is reserved. The Parameter2 field (Time2Wait) indicates,
102      * in seconds, the minimum time to wait before attempting to reconnect. The
103      * Parameter3 field (Time2Retain) indicates the maximum time allowed to
104      * reassign commands after the initial wait (in Parameter2). If the
105      * initiator does not attempt to reconnect and/or reassign the outstanding
106      * commands within the time specified by Parameter3, or if Parameter3 is 0,
107      * the session is terminated. In this case, the target will terminate all
108      * outstanding commands in this session; no other responses should be
109      * expected from the target for the outstanding commands in this session. A
110      * value of 0 for Parameter2 indicates that reconnect can be attempted
111      * immediately.</td>
112      * </tr>
113      * <tr>
114      * <td>4</td>
115      * <td>Target requests parameter negotiation on this connection. The
116      * initiator MUST honor this request by issuing a Text Request (that can be
117      * empty) on the same connection as early as possible, but no later than
118      * Parameter3 seconds, unless a Text Request is already pending on the
119      * connection, or by issuing a Logout Request. If the initiator does not
120      * issue a Text Request the target may reissue the Asynchronous Message
121      * requesting parameter negotiation.</td>
122      * </tr>
123      * <tr>
124      * <td>255</td>
125      * <td>Vendor specific iSCSI Event. The AsyncVCode details the vendor code,
126      * and data MAY accompany the report.</td>
127      * </tr>
128      * </table>
129      * <br/>
130      * All other event codes are reserved.
131      */
132     public static enum AsyncEventCodes {
133         /** A SCSI Asynchronous Event is reported in the sense data. */
134         SCSI_ASYNCHRONOUS_EVENT((byte) 0),
135 
136         /** Target requests Logout. */
137         LOGOUT((byte) 1),
138 
139         /** Target indicates it will drop the connection. */
140         DROP_CONNECTION((byte) 2),
141 
142         /** Target indicates it will drop all the connections of this session. */
143         DROP_ALL_CONNECTIONS((byte) 3),
144 
145         /** Target requests parameter negotiation on this connection. */
146         PARAMETER_NEGOTIATION((byte) 4),
147 
148         /** Vendor specific iSCSI Event. */
149         VENDOR_SPECIFIC_EVENT((byte) 255);

330     /** {@inheritDoc} */
331     @Override
332     protected final void deserializeBytes36to39(final int line)
333             throws InternetSCSIException {
334 
335         asyncEvent = AsyncEventCodes
336                 .valueOf((byte) (line & Constants.FIRST_BYTE_MASK));
337         asyncVCode = (byte) (line & Constants.SECOND_BYTE_MASK);
338         parameter1 = (short) (line & Constants.LAST_TWO_BYTES_MASK);
339     }
340 
341     /** {@inheritDoc} */
342     @Override
343     protected final void deserializeBytes40to43(final int line)
344             throws InternetSCSIException {
345 
346         parameter2 = (short) (line & Constants.FIRST_TWO_BYTES_MASK);
347         parameter3 = (short) (line & Constants.LAST_TWO_BYTES_MASK);
348     }

353     /** {@inheritDoc} */
354     @Override
355     protected final void checkIntegrity() throws InternetSCSIException {
356 
357         String exceptionMessage;
358         do {
359             if (asyncVCode != 0
360                     && asyncEvent != AsyncEventCodes.VENDOR_SPECIFIC_EVENT) {
361                 exceptionMessage = "A vendor specific code is only valid, if also a vendor specific event occured.";
362                 break;
363             }
364 
365             if (asyncEvent == AsyncEventCodes.SCSI_ASYNCHRONOUS_EVENT
366                     && logicalUnitNumber == 0) {
367                 exceptionMessage = "A valid LogicalUnitNumber must be given, if an asynchronous event occured.";
368                 break;
369             } else {
370                 Utils.isReserved(logicalUnitNumber);
371             }
372             // message is checked correctly
373             return;
374         } while (false);
375 
376         throw new InternetSCSIException(exceptionMessage);
377     }

383     @Override
384     protected final int serializeBytes36to39() {
385 
386         int line = parameter1;
387         line |= asyncVCode << Constants.TWO_BYTES_SHIFT;
388         line |= asyncEvent.value() << Constants.THREE_BYTES_SHIFT;
389 
390         return line;
391     }
