32  /**
33   * <h1>OperationCode</h1>
34   * <p>
35   * This enumeration defines all valid operation codes, which are conform to
36   * iSCSI Protocol (RFC 3720).
37   * <p>
38   * <table border="1">
39   * <tr>
40   * <td>0x00</td>
41   * <td>NOP-Out</td>
42   * </tr>
43   * <tr>
44   * <td>0x01</td>
45   * <td>SCSI Command (encapsulates a SCSI Command Descriptor Block)</td>
46   * </tr>
47   * <tr>
48   * <td>0x02</td>
49   * <td>SCSI Task Management function request</td>
50   * </tr>
51   * <tr>
52   * <td>0x03</td>
53   * <td>Login Request</td>
54   * </tr>
55   * <tr>
56   * <td>0x04</td>
57   * <td>Text Request</td>
58   * </tr>
59   * <tr>
60   * <td>0x05</td>
61   * <td>SCSI Data-Out (for WRITE operations)</td>
62   * </tr>
63   * <tr>
64   * <td>0x06</td>
65   * <td>Logout Request</td>
66   * </tr>
67   * <tr>
68   * <td>0x10</td>
69   * <td>SNACK Request</td>
70   * </tr>
71   * <tr>
72   * <td>0x1c-0x1e</td>
73   * <td>Vendor specific codes</td>
74   * </tr>
75   * <tr>
76   * <td>0x20</td>
77   * <td>NOP-In</td>
78   * </tr>
79   * <tr>
80   * <td>0x21</td>
81   * <td>SCSI Response - contains SCSI status and possibly sense information or
82   * other response information.</td>
83   * </tr>
84   * <tr>
85   * <td>0x22</td>
86   * <td>SCSI Task Management function response</td>
87   * </tr>
88   * <tr>
89   * <td>0x23</td>
90   * <td>Login Response</td>
91   * </tr>
92   * <tr>
93   * <td>0x24</td>
94   * <td>Text Response</td>
95   * </tr>
96   * <tr>
97   * <td>0x25</td>
98   * <td>SCSI Data-In - for READ operations.</td>
99   * </tr>
100  * <tr>
101  * <td>0x26</td>
102  * <td>Logout Response</td>
103  * </tr>
104  * <tr>
105  * <td>0x31</td>
106  * <td>Ready To Transfer (R2T) - sent by target when it is ready to receive
107  * data.</td>
108  * </tr>
109  * <tr>
110  * <td>0x32</td>
111  * <td>Asynchronous Message - sent by target to indicate certain special
112  * conditions.</td>
113  * </tr>
114  * <tr>
115  * <td>0x3c-0x3e</td>
116  * <td>Vendor specific codes</td>
117  * </tr>
118  * <tr>
119  * <td>0x3f</td>
120  * <td>Reject</td>
121  * </tr>
122  * </table>
123  * <p>
124  * Not Supported: 0x1c-0x1e Vendor specific codes
125  * 
126  * @author Volker Wildi
127  */
128 public enum OperationCode {
129 
130     // --------------------------------------------------------------------------
131     // The initiator operation codes
132     // --------------------------------------------------------------------------
133 
134     /**
135      * This request/response pair may be used by an initiator and target as a
136      * "ping" mechanism to verify that a connection/session is still active and
137      * all of its components are operational. Such a ping may be triggered by
138      * the initiator or target. The triggering party indicates that it wants a
139      * reply by setting a value different from the default
140      * <code>0xffffffff</code> in the corresponding Initiator/Target Transfer
141      * Tag. NOP-In/NOP-Out may also be used "unidirectional" to convey to the
142      * initiator/target command, status or data counter values when there is no
143      * other "carrier" and there is a need to update the initiator/ target.
144      */
145     NOP_OUT((byte) 0x00),
146 
147     /**
148      * This request carries the SCSI CDB and all the other SCSI execute command
149      * procedure call (see [SAM2]) IN arguments such as task attributes,
150      * Expected Data Transfer Length for one or both transfer directions (the
151      * latter for bidirectional commands), and Task Tag (as part of the I_T_L_x
152      * nexus). The I_T_L nexus is derived by the initiator and target from the
153      * LUN field in the request and the I_T nexus is implicit in the session
154      * identification. In addition, the SCSI-command PDU carries information
155      * required for the proper operation of the iSCSI protocol - the command
156      * sequence number ( <code>CmdSN</code>) for the session and the expected
157      * status number ( <code>ExpStatSN</code>) for the connection. All or part
158      * of the SCSI output (write) data associated with the SCSI command may be
159      * sent as part of the SCSI-Command PDU as a data segment.
160      */
161     SCSI_COMMAND((byte) 0x01),
162 
163     /**
164      * The Task Management function request provides an initiator with a way to
165      * explicitly control the execution of one or more SCSI Tasks or iSCSI
166      * functions. The PDU carries a function identifier (which task management
167      * function to perform) and enough information to unequivocally identify the
168      * task or task-set on which to perform the action, even if the task(s) to
169      * act upon has not yet arrived or has been discarded due to an error. The
170      * referenced tag identifies an individual task if the function refers to an
171      * individual task. The I_T_L nexus identifies task sets. In iSCSI the I_T_L
172      * nexus is identified by the LUN and the session identification (the
173      * session identifies an I_T nexus). For task sets, the <code>CmdSN</code>
174      * of the Task Management function request helps identify the tasks upon
175      * which to act, namely all tasks associated with a LUN and having a CmdSN
176      * preceding the Task Management function request <code>CmdSN</code>. For a
177      * Task Management function, the coordination between responses to the tasks
178      * affected and the Task Management function response is done by the target.
179      */
180     SCSI_TM_REQUEST((byte) 0x02),
181 
182     /**
183      * Login Requests and Responses are used exclusively during the Login Phase
184      * of each connection to set up the session and connection parameters. (The
185      * Login Phase consists of a sequence of login requests and responses
186      * carrying the same Initiator Task Tag.) A connection is identified by an
187      * arbitrarily selected connection-ID (<code>CID</code>) that is unique
188      * within a session. Similar to the Text Requests and Responses, Login
189      * Requests/Responses carry key=value text information with a simple syntax
190      * in the data segment. The Login Phase proceeds through several stages
191      * (security negotiation, operational parameter negotiation) that are
192      * selected with two binary coded fields in the header -- the
193      * "current stage" (<code>CSG</code>) and the "next stage" (<code>NSG</code>
194      * ) with the appearance of the latter being signaled by the "transit" flag
195      * (<code>T</code>). The first Login Phase of a session plays a special
196      * role, called the leading login, which determines some header fields
197      * (e.g., the version number, the maximum number of connections, and the
198      * session identification). The CmdSN initial value is also set by the
199      * leading login. StatSN for each connection is initiated by the connection
200      * login. A login request may indicate an implied logout (cleanup) of the
201      * connection to be logged in (a connection restart) by using the same
202      * Connection ID (CID) as an existing connection, as well as the same
203      * session identifying elements of the session to which the old connection
204      * was associated.
205      */
206     LOGIN_REQUEST((byte) 0x03),
207 
208     /**
209      * Text requests and responses are designed as a parameter negotiation
210      * vehicle and as a vehicle for future extension. In the data segment, Text
211      * Requests/Responses carry text information using a simple "key=value"
212      * syntax. Text Request/Responses may form extended sequences using the same
213      * Initiator Task Tag. The initiator uses the F (Final) flag bit in the text
214      * request header to indicate its readiness to terminate a sequence. The
215      * target uses the F (Final) flag bit in the text response header to
216      * indicate its consent to sequence termination. Text Request and Responses
217      * also use the Target Transfer Tag to indicate continuation of an operation
218      * or a new beginning. A target that wishes to continue an operation will
219      * set the Target Transfer Tag in a Text Response to a value different from
220      * the default 0xffffffff. An initiator willing to continue will copy this
221      * value into the Target Transfer Tag of the next Text Request. If the
222      * initiator wants to restart the current target negotiation (start fresh)
223      * will set the Target Transfer Tag to 0xffffffff. Although a complete
224      * exchange is always started by the initiator, specific parameter
225      * negotiations may be initiated by the initiator or target.
226      */
227     TEXT_REQUEST((byte) 0x04),
228 
229     /**
230      * SCSI Data-Out and SCSI Data-In are the main vehicles by which SCSI data
231      * payload is carried between initiator and target. Data payload is
232      * associated with a specific SCSI command through the Initiator Task Tag.
233      * For target convenience, outgoing solicited data also carries a Target
234      * Transfer Tag (copied from R2T) and the LUN. Each PDU contains the payload
235      * length and the data offset relative to the buffer address contained in
236      * the SCSI execute command procedure call. In each direction, the data
237      * transfer is split into "sequences". An end-of-sequence is indicated by
238      * the F bit. An outgoing sequence is either unsolicited (only the first
239      * sequence can be unsolicited) or consists of all the Data-Out PDUs sent in
240      * response to an R2T. Input sequences are built to enable the direction
241      * switching for bidirectional commands. For input, the target may request
242      * positive acknowledgement of input data. This is limited to sessions that
243      * support error recovery and is implemented through the A bit in the SCSI
244      * Data-In PDU header. Data-In and Data-Out PDUs also carry the DataSN to
245      * enable the initiator and target to detect missing PDUs (discarded due to
246      * an error). In addition, StatSN is carried by the Data-In PDUs. To enable
247      * a SCSI command to be processed while involving a minimum number of
248      * messages, the last SCSI Data-In PDU passed for a command may also contain
249      * the status if the status indicates termination with no exceptions (no
250      * sense or response involved).
251      */
252     SCSI_DATA_OUT((byte) 0x05),
253 
254     /**
255      * Logout Requests and Responses are used for the orderly closing of
256      * connections for recovery or maintenance. The logout request may be issued
257      * following a target prompt (through an asynchronous message) or at an
258      * initiators initiative. When issued on the connection to be logged out, no
259      * other request may follow it.
260      */
261     LOGOUT_REQUEST((byte) 0x06),
262 
263     /**
264      * With the SNACK Request, the initiator requests retransmission of
265      * numbered-responses or data from the target. A single SNACK request covers
266      * a contiguous set of missing items, called a run, of a given type of
267      * items. The type is indicated in a type field in the PDU header. The run
268      * is composed of an initial item (StatSN, DataSN, R2TSN) and the number of
269      * missed Status, Data, or R2T PDUs. For long Data-In sequences, the target
270      * may request (at predefined minimum intervals) a positive acknowledgement
271      * for the data sent. A SNACK request with a type field that indicates ACK
272      * and the number of Data-In PDUs acknowledged conveys this positive
273      * acknowledgement.
274      */
275     SNACK_REQUEST((byte) 0x10),

281     /**
282      * This request/response pair may be used by an initiator and target as a
283      * "ping" mechanism to verify that a connection/session is still active and
284      * all of its components are operational. Such a ping may be triggered by
285      * the initiator or target. The triggering party indicates that it wants a
286      * reply by setting a value different from the default 0xffffffff in the
287      * corresponding Initiator/Target Transfer Tag. NOP-In/NOP-Out may also be
288      * used "unidirectional" to convey to the initiator/target command, status
289      * or data counter values when there is no other "carrier" and there is a
290      * need to update the initiator/ target.
291      */
292     NOP_IN((byte) 0x20),
293 
294     /**
295      * The SCSI-Response carries all the SCSI execute-command procedure call
296      * (see [SAM2]) OUT arguments and the SCSI execute-command procedure call
297      * return value. The SCSI-Response contains the residual counts from the
298      * operation, if any, an indication of whether the counts represent an
299      * overflow or an underflow, and the SCSI status if the status is valid or a
300      * response code (a non-zero return value for the execute-command procedure
301      * call) if the status is not valid. For a valid status that indicates that
302      * the command has been processed, but resulted in an exception (e.g., a
303      * SCSI CHECK CONDITION), the PDU data segment contains the associated sense
304      * data. The use of Autosense ([SAM2]) is REQUIRED by iSCSI. Some data
305      * segment content may also be associated (in the data segment) with a
306      * non-zero response code. In addition, the SCSI-Response PDU carries
307      * information required for the proper operation of the iSCSI protocol:
308      * <ul>
309      * <li>The number of Data-In PDUs that a target has sent (to enable the
310      * initiator to check that all have arrived).</li>
311      * <li>StatSN - the Status Sequence Number on this connection</li>
312      * <li>ExpCmdSN - the next Expected Command Sequence Number at the target.</li>
313      * <li>MaxCmdSN - the maximum CmdSN acceptable at the target from this
314      * initiator.</li>
315      * </ul>
316      */
317     SCSI_RESPONSE((byte) 0x21),
318 
319     /**
320      * The Task Management function response carries an indication of function
321      * completion for a Task Management function request including how it was
322      * completed (response and qualifier) and additional information for failure
323      * responses. After the Task Management response indicates Task Management
324      * function completion, the initiator will not receive any additional
325      * responses from the affected tasks.
326      */
327     SCSI_TM_RESPONSE((byte) 0x22),
328

339      * selected with two binary coded fields in the header -- the

342      * (<code>T</code>). The first Login Phase of a session plays a special

373      * initiated by the initiator or target.

402     /**
403      * Logout Requests and Responses are used for the orderly closing of
404      * connections for recovery or maintenance. The Logout Response indicates
405      * that the connection or session cleanup is completed and no other
406      * responses will arrive on the connection (if received on the logging out
407      * connection). In addition, the Logout Response indicates how long the
408      * target will continue to hold resources for recovery (e.g., command

412      */
413     LOGOUT_RESPONSE((byte) 0x26),

415     /**
416      * R2T is the mechanism by which the SCSI target "requests" the initiator
417      * for output data. R2T specifies to the initiator the offset of the
418      * requested data relative to the buffer address from the execute command
419      * procedure call and the length of the solicited data. To help the SCSI
420      * target associate the resulting Data-Out with an R2T, the R2T carries a
421      * Target Transfer Tag that will be copied by the initiator in the solicited
422      * SCSI Data-Out PDUs. There are no protocol specific requirements with
423      * regard to the value of these tags, but it is assumed that together with
424      * the LUN, they will enable the target to associate data with an R2T. R2T
425      * also carries information required for proper operation of the iSCSI
426      * protocol, such as:
427      * <ul>
428      * <li>R2TSN (to enable an initiator to detect a missing R2T)</li>
429      * <li>StatSN</li>
430      * <li>ExpCmdSN</li>
431      * <li>MaxCmdSN</li>
432      * </ul>
433      */
434     R2T((byte) 0x31),
