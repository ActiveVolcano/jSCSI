36  /**
37   * <h1>LoginResponseParser</h1>
38   * <p>
39   * This is a parser for the Login Response Message of the iSCSI Protocol (RFC3720). So it parses all the field
40   * of this login message and offers it fields with its getter methods.
41   * <p>
42   * The Login Response indicates the progress and/or end of the Login Phase.
43   * <p>
44   * <h4>T (Transit) bit</h4> The <code>T</code> bit is set to <code>1</code> as an indicator of the end of the
45   * stage. If the <code>T</code> bit is set to <code>1</code> and NSG is FullFeaturePhase, then this is also
46   * the Final Login Response (see Chapter 5). A <code>T</code> bit of <code>0</code> indicates a "partial"
47   * response, which means "more negotiation needed".
48   * <p>
49   * A Login Response with a <code>T</code> bit set to <code>1</code> MUST NOT contain key=value pairs that may
50   * require additional answers from the initiator within the same stage.
51   * <p>
52   * A Login Response with a <code>T</code> bit set to <code>1</code> MUST NOT contain key=value pairs that may
53   * require additional answers from the initiator within the same stage.
54   * <p>
55   * <h4>StatSN</h4> For the first Login Response (the response to the first Login Request), this is the
56   * starting status Sequence Number for the connection. The next response of any kind, including the next Login
57   * Response, if any, in the same Login Phase, will carry this <code>number + 1</code>. This field is only
58   * valid if the Status-Class is <code>0</code>.
59   * <p>
60   * <h4>Login Parameters</h4> The target MUST provide some basic parameters in order to enable the initiator to
61   * determine if it is connected to the correct port and the initial text parameters for the security exchange.
62   * <p>
63   * All the rules specified in Section 10.11.6 Text Response Data for text responses also hold for Login
64   * Responses. Keys and their explanations are listed in Chapter 11 (security negotiation keys) and Chapter 12
65   * (operational parameter negotiation keys). All keys in Chapter 12, except for the X extension formats, MUST
66   * be supported by iSCSI initiators and targets. Keys in Chapter 11, only need to be supported when the
67   * function to which they refer is mandatory to implement.
68   * 
69   * @author Volker Wildi
70   */

80       * The Status returned in a Login Response indicates the execution status of
81       * the Login Phase. The status includes:
82       * <ul>
83       * <li>Status-Class</li>
84       * <li>Status-Detail</li>
85       * </ul>
86       * <p>
87       * <code>0</code> Status-Class indicates success.
88       * <p>
89       * A non-zero Status-Class indicates an exception. In this case, Status-Class is sufficient for a simple
90       * initiator to use when handling exceptions, without having to look at the Status-Detail. The
91       * Status-Detail allows finer-grained exception handling for more sophisticated initiators and for better
92       * information for logging.
93       * <p>
94       * If the Status Class is not <code>0</code>, the initiator and target MUST close the TCP connection.
95       * <p>
96       * If the target wishes to reject the Login Request for more than one reason, it should return the primary
97       * reason for the rejection.
98       * 
99       * @see LoginStatus
100      */

342     /** {@inheritDoc} */
343     @Override
344     protected final void deserializeBytes1to3(final int line) throws InternetSCSIException {
345 
346         continueFlag = Utils.isBitSet(line & Constants.CONTINUE_FLAG_MASK);
347 
348         Utils.isReserved(line & LoginConstants.BIT_11_AND_12_FLAG_MASK);
349 
350         currentStageNumber =
351             LoginStage
352                 .valueOf((byte)((line & LoginConstants.CSG_FLAG_MASK) >>> LoginConstants.CSG_BIT_SHIFT));
353         nextStageNumber =
354             LoginStage.valueOf((byte)((line & LoginConstants.NSG_FLAG_MASK) >>> Constants.TWO_BYTES_SHIFT));
355 
356         maxVersion = (line & Constants.THIRD_BYTE_MASK) >> Constants.ONE_BYTE_SHIFT;
357         activeVersion = line & Constants.FOURTH_BYTE_MASK;
358     }
359 
360     /** {@inheritDoc} */
361     @Override
362     protected final void deserializeBytes12to15(final int line) throws InternetSCSIException {
363 
364         // use the logicalUnitNumber variable as temporary storage
365         logicalUnitNumber |= Utils.getUnsignedLong(line);
366         initiatorSessionID.deserialize(logicalUnitNumber);
367 
368         targetSessionIdentifyingHandle = (short)(line & Constants.LAST_TWO_BYTES_MASK);
369     }
370 
371     /** {@inheritDoc} */
372     @Override
373     protected final void deserializeBytes36to39(final int line) throws InternetSCSIException {
374 
375         status =
376             LoginStatus
377                 .valueOf((short)((line & Constants.FIRST_TWO_BYTES_MASK) >>> Constants.TWO_BYTES_SHIFT));
378         Utils.isReserved(line & Constants.LAST_TWO_BYTES_MASK);
379     }

384     /** {@inheritDoc} */
385     @Override
386     protected final void checkIntegrity() throws InternetSCSIException {
387 
388         String exceptionMessage;
389         do {
390             if (status != LoginStatus.SUCCESS && statusSequenceNumber != 0) {
391                 exceptionMessage =
392                     "While no successful login is preformed, the StatusSequenceNumber must be 0.";
393                 break;
394             }
395 
396             // message is checked correctly
397             return;
398         } while (false);
399 
400         throw new InternetSCSIException(exceptionMessage);
401     }

406     /** {@inheritDoc} */
407     @Override
408     protected final int serializeBytes1to3() {
409 
410         int line = activeVersion;
411         line |= maxVersion << Constants.ONE_BYTE_SHIFT;
412         line |= nextStageNumber.value() << Constants.TWO_BYTES_SHIFT;
413         line |= currentStageNumber.value() << LoginConstants.CSG_BIT_SHIFT;
414 
415         if (continueFlag) {
416             line |= Constants.CONTINUE_FLAG_MASK;
417         }
418         return line;
419     }
420 
421     /** {@inheritDoc} */
422     @Override
423     protected final int serializeBytes36to39() {
424 
425         int line = 0;
426         line |= status.value() << Constants.TWO_BYTES_SHIFT;
427 
428         return line;
429     }
