14   * Instances of this class can be used for persistent storage of data. They are
15   * backed by a {@link RandomAccessFile}, which will immediately write all
16   * changes in the data to hard-disk.

116     /**
117      * This is the build method for creating instances of {@link RandomAccessStorageModule}. If there is no
118      * file to be found at the
119      * specified <code>filePath</code>, then a {@link FileNotFoundException} will be thrown.
120      * 
121      * @param file
122      *            a path leading to the file serving as storage medium
123      * @param storageLength
124      *            length of storage (if not already existing)
125      * @param create
126      *            should the storage be created
127      * @return a new instance of {@link RandomAccessStorageModule}
128      * @throws IOException
129      */
130     public static synchronized final IStorageModule open(final File file, final long storageLength,
131         final boolean create, IStorageModule.STORAGEKIND kind) throws IOException {
132         long sizeInBlocks;
133         RandomAccessFile randomAccessFile;
134         if (create) {
135             sizeInBlocks = storageLength / VIRTUAL_BLOCK_SIZE;
136             createStorageVolume(file, storageLength);
137         } else {
138             sizeInBlocks = file.length() / VIRTUAL_BLOCK_SIZE;
139         }
140         // throws exc. if !file.exists()
141         randomAccessFile = new RandomAccessFile(file, MODE);
142         switch (kind) {
143         case SyncFile:
144             return new SynchronizedRandomAccessStorageModule(sizeInBlocks, randomAccessFile);
145         case AsyncFile:
146             return new RandomAccessStorageModule(sizeInBlocks, randomAccessFile);
147         }
148         throw new IllegalStateException("No kind found, no storage initialized");
149 
150     }

152     /**
153      * Creating a new file if not existing at the path defined in the config.
154      * Note that it is advised to create the file beforehand.
155      * 
156      * @param pConf
157      *            configuration to be updated
158      * @return true if creation successful, false if file already exists.
159      * @throws IOException
160      *             if anything weird happens
161      */
162     private static synchronized boolean createStorageVolume(final File pToCreate, final long pLength)
163         throws IOException {
164         FileOutputStream outStream = null;
165         try {
166             // if file exists, remove it after questioning.
167             if (pToCreate.exists()) {
168                 if (!pToCreate.delete()) {
169                     LOGGER.debug("Removal of old storage " + pToCreate.toString() + " unsucessful.");
170                     return false;
171                 }
172                 LOGGER.debug("Removal of old storage " + pToCreate.toString() + " sucessful.");
173             }
174 
175             // create file
176             final File parent = pToCreate.getCanonicalFile().getParentFile();
177             if (!parent.exists() && !parent.mkdirs()) {
178                 throw new FileNotFoundException("Unable to create directory: " + parent.getAbsolutePath());
179             }
180 
181             pToCreate.createNewFile();
182             outStream = new FileOutputStream(pToCreate);
183             final FileChannel fcout = outStream.getChannel();
184             fcout.position(pLength);
185             outStream.write(26); // Write EOF (not normally needed)
186             fcout.force(true);
187             LOGGER.debug("Creation of storage " + pToCreate.toString() + " sucessful.");
188             return true;
189         } catch (IOException e) {
190             LOGGER.error("Exception creating storage volume " + pToCreate.getAbsolutePath() + ": "
191                 + e.getMessage(), e);
192             throw e;
193         } finally {
194             if (outStream != null) {
195                 try {
196                     outStream.close();
197                 } catch (IOException e) {
198                     LOGGER.error("Exception closing storage volume: " + e.getMessage(), e);
199                 }
200             }
201         }
202 
203     }
