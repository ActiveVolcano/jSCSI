3   import java.util.concurrent.locks.Lock;
4   import java.util.concurrent.locks.ReentrantLock;
5   
6   /**
7    * Instances of {@link BinaryLock} can be used to prevent concurrent access to
8    * the same resource, so, in essence, this is a very simplified {@link Lock} implementation, however lacking
9    * many advanced capabilities.
10   * <p>
11   * A {@link BinaryLock} knows only two states, locked and unlocked. Attempts by the lock-holder to lock a
12   * {@link BinaryLock} when locked or to unlock it when unlocked, will have no effect.
13   * 
14   * @author Andreas Ergenzinger
15   */
16  public class BinaryLock {
17  
18      /**
19       * The {@link ReentrantLock} which backs up the {@link BinaryLock} and takes
20       * care of suspending and notifying waiting {@link Threads}.
21       */
22      private final ReentrantLock lock = new ReentrantLock();
23  
24      /**
25       * This method is used to acquire the lock. It will block until no other {@link Thread} is holding the
26       * lock and then return <code>true</code> to
27       * indicate the successful lock acquisition, or return <code>false</code>,
28       * if the calling {@link Thread} was interrupted while waiting for the lock.
29       * <p>
30       * If the caller is already holding the lock, the method will immediately return <code>true</code> without
31       * any changes.
32       * 
33       * @return <code>true</code> if and only if the lock has been acquired
34       */
35      public boolean lock() {
36          try {
37              lock.lockInterruptibly();
38              return true;
39          } catch (InterruptedException e) {
40              return false;
41          }
42      }
43  
44      /**
45       * Releases the lock when called by the current lock holder;
46       */
47      public void unlock() {
48          if (lock.isHeldByCurrentThread()) {
49              while (lock.getHoldCount() > 0)
50                  lock.unlock();
51          }
52      }

54  }
