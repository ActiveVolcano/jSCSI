5   /**
6    * {@link BooleanResultFunction}s are used by instances of{@link BooleanEntry} during the negotiation of
7    * boolean parameters.
8    * <p>
9    * They determine a negotiation outcome based on a logical <code>AND</code> or <code>OR</code> operation.
10   * 
11   * @author Andreas Ergenzinger
12   */
13  public enum BooleanResultFunction {
14      /**
15       * The negotiation result will be <code>true</code> only if both initiator
16       * and target support that value.
17       */
18      AND,
19      /**
20       * The negotiation result will be <code>true</code> only if either the
21       * initiator or the target support that value.
22       */
23      OR;
24  
25      /**
26       * Performs a logical <code>AND</code> or <code>OR</code> operation on the
27       * two parameters and returns the result. The type of the operation depends
28       * on the value of this {@link BooleanResultFunction}.
29       * 
30       * @param a
31       *            the first boolean value
32       * @param b
33       *            the second boolean value
34       * @return the result of a logical <code>AND</code> or <code>OR</code> operation.
35       */
36      public final boolean getResult(final boolean a, final boolean b) {
37          if (this == AND)
38              return a && b;
39          return a || b;
40      }
