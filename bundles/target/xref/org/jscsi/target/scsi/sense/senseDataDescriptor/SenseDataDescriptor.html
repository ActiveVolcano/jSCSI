7   /**
8    * Sense data descriptors are part of sense data and provide specific sense
9    * information. A given type of sense data descriptor shall be included in the
10   * sense data only when the information it contains is valid.
11   * 
12   * @author Andreas Ergenzinger
13   */
14  public abstract class SenseDataDescriptor implements ISerializable {

16      /**
17       * This value plus the length of all sub-class-specific fields equal the
18       * length in bytes of the serialized sense data descriptor.
19       */
20      private static final int COMMON_FIELDS_LENGTH = 2;
21  
22      /**
23       * The descriptor type field determines sense data descriptor's
24       * (non-abstract) class.
25       * 
26       * @see SenseDataDescriptorType
27       */
28      private final SenseDataDescriptorType descriptorType;
29  
30      /**
31       * The ADDITIONAL LENGTH field indicates the number of sense data descriptor
32       * specific bytes that follow in the sense data descriptor.
33       */
34      private final int additionalLength;
35  
36      /**
37       * The abstract constructor.
38       * 
39       * @param descriptorType
40       *            determines the specific type of the sense data descriptor
41       * @param additionalLength
42       *            the length in bytes of all additional fields
43       * @see #COMMON_FIELDS_LENGTH
44       */
45      public SenseDataDescriptor(final SenseDataDescriptorType descriptorType, final int additionalLength) {
46          this.descriptorType = descriptorType;
47          this.additionalLength = additionalLength;
48      }
49  
50      /**
51       * Serializes the fields common to all sense data descriptors.
52       * 
53       * @param byteBuffer
54       *            where the serialized fields will be stored
55       * @param index
56       *            the position of the first byte of the sense data descriptor in
57       *            the {@link ByteBuffer}
58       */
59      private final void serializeCommonFields(final ByteBuffer byteBuffer, final int index) {
60          byteBuffer.position(index);
61          byteBuffer.put(descriptorType.getValue());
62          byteBuffer.put((byte)additionalLength);
63      }
64  
65      /**
66       * Serializes all fields which are not common to all sense data descriptors,
67       * which means those that are sub-type-specific.

71       * @param index
72       *            the position of the first byte of the sense data descriptor in
73       *            the {@link ByteBuffer}
74       */
75      protected abstract void serializeSpecificFields(ByteBuffer byteBuffer, final int index);
76  
77      private final int getAdditionalLength() {
78          return additionalLength;
79      }
80  
81      public void serialize(ByteBuffer byteBuffer, int index) {
82          serializeCommonFields(byteBuffer, index);
83          serializeSpecificFields(byteBuffer, index);
84      }
85  
86      public int size() {
