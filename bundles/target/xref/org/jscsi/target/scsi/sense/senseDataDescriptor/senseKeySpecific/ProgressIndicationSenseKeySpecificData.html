8    * Progress indication sense-key-specific data is used to communicate the

14

16       * The byte position of the PROGRESS INDICATION field.
17       */
18      private static final int PROGRESS_INDICATION_FIELD_INDEX = 1;
19  
20      /**
21       * The PROGRESS INDICATION field is a percent complete indication in which
22       * the returned value is a numerator that has 65 536 (10000h) as its
23       * denominator. The progress indication shall be based upon the total
24       * operation.
25       * <p>
26       * The progress indication should be time related, however this is not an absolute requirement. (E.g.,
27       * since format time varies with the number of defects encountered, etc., it is reasonable for the device
28       * server to assign values to various steps within the process. The granularity of these steps should be
29       * small enough to provide reasonable assurances to the application client that progress is being made.)
30       */
31      private final short progressIndication;
32  
33      /**
34       * The consctructor.
35       * 
36       * @param senseKeySpecificDataValid
37       *            <code>true</code> if and only if the <i>progressIndication</i>
38       *            parameter is to be considered valid
39       * @param progressIndication
40       *            specifies the progress, the integer must lie in the interval
41       *            [0, 65536]
42       */
43      public ProgressIndicationSenseKeySpecificData(final boolean senseKeySpecificDataValid,
44          final int progressIndication) {
45          super(senseKeySpecificDataValid);
46          this.progressIndication = (short)progressIndication;
47      }
48  
49      @Override
50      protected void serializeSpecificFields(ByteBuffer byteBuffer, int index) {
