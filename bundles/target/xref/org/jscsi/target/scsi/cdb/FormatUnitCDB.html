7   /**
8    * This class represents Command Descriptor Blocks for the <code>FORMAT UNIT</code> SCSI command.
9    * 
10   * @author Andreas Ergenzinger
11   */
12  public final class FormatUnitCDB extends CommandDescriptorBlock {
13  
14      /**
15       * The format protection information (FMTPINFO) field in combination with
16       * the PROTECTION FIELD USAGE field
17       */
18      private byte formatProtectionInformation;
19  
20      /**
21       * A LONGLIST bit set to zero specifies that the parameter list, if any,
22       * contains a short parameter list header as defined in SBC-3. A LONGLIST
23       * bit set to one specifies that the parameter list, if any, contains a long
24       * parameter list header.
25       * <p>
26       * If the FMTDATA bit is set to zero, then the LONGLIST bit shall be ignored.
27       */
28      private boolean longList;
29  
30      /**
31       * A format data (FMTDATA) bit set to zero specifies that no parameter list
32       * be transferred from the data-out buffer (i.e. the PDU's data segment is
33       * empty).
34       */
35      private boolean formatData;
36  
37      /**
38       * A complete list (CMPLST) bit set to zero specifies that the defect list
39       * included in the FORMAT UNIT parameter list shall be used in an addition
40       * to the existing list of defects.
41       */
42      private boolean completeList;
43  
44      /**
45       * If the FMTDATA bit is set to one, then the DEFECT LIST FORMAT field
46       * specifies the format of the address descriptors in the defect list.
47       */
48      private DefectListFormat defectListFormat;
49  
50      public FormatUnitCDB(ByteBuffer buffer) {
51          super(buffer);
52  
53          byte b = buffer.get(1);
54  
55          // format protection information
56          formatProtectionInformation = (byte)(b >>> 6);
57  
58          // long list
59          longList = BitManip.getBit(b, 5);
60  
61          // format data
62          formatData = BitManip.getBit(b, 4);
63  
64          // complete list
65          completeList = BitManip.getBit(b, 3);
66  
67          // defect list format
68          defectListFormat = DefectListFormat.valueOf(b & 7);
69          if (!formatData) {
70              if (defectListFormat != DefectListFormat.SHORT_BLOCK)
71                  addIllegalFieldPointer(1, 2);// illegal defect list format
72              /*
73               * If the FMTDATA bit is set to zero and the FMTPINFO field is not
74               * set to zero, then the device server shall terminate the command
75               * with CHECK CONDITION status with the sense key set to ILLEGAL
76               * REQUEST and the additional sense code set to INVALID FIELD IN
77               * CDB.
78               */
79              if (formatProtectionInformation != 0)
80                  addIllegalFieldPointer(1, 7);
81          } else if (formatData)
82              addIllegalFieldPointer(1, 4);// no support for format data
83      }
84  
85      public final boolean getCompleteList() {
86          return completeList;
87      }
88  
89      public final int getFormatProtectionInformation() {
90          return formatProtectionInformation;
91      }
92  
93      public final boolean getLongList() {
94          return longList;
95      }
