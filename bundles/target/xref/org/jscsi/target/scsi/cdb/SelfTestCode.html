32      BACKGROUND_EXTENDED_SELF_TEST((byte)2),

43      ABORT_BACKGROUND_SELF_TEST((byte)4),

49      FOREGROUND_SELF_TEST((byte)5),
50      /**

52       * mode. The {@link SendDiagnosticCdb#parameterListLength} field shall

55      FOREGROUND_EXTENDED_SELF_TEST((byte)6);
