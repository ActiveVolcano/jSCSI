6    * This is an abstract super-class for command descriptor blocks that that are
7    * used for requesting information about the capacity of the storage device.
8    * 
9    * @author Andreas Ergenzinger
10   */
11  public abstract class ReadCapacityCdb extends CommandDescriptorBlock {

13      /**
14       * The LOGICAL BLOCK ADDRESS field specifies the LBA of the first logical
15       * block accessed by this command.
16       * <p>
17       * If the specified LBA exceeds the capacity of the medium, then the device server shall terminate the
18       * command with CHECK CONDITION status with the sense key set to ILLEGAL REQUEST and the additional sense
19       * code set to LOGICAL BLOCK ADDRESS OUT OF RANGE.
20       * 
21       * @see #partialMediumIndicator
22       */
23      protected final long logicalBlockAddress;
24  
25      /**
26       * A partial medium indicator (PMI) bit set to zero specifies that the
27       * device server return information on the last logical block on the
28       * direct-access block device.
29       * <p>
30       * A PMI bit set to one specifies that the device server return information on the last logical block
31       * after that specified in the LOGICAL BLOCK ADDRESS field before a substantial vendor specific delay in
32       * data transfer may be encountered.
33       * <p>
34       * The LOGICAL BLOCK ADDRESS field shall be set to zero if the PMI bit is set to zero. If the PMI bit is
35       * set to zero and the LOGICAL BLOCK ADDRESS field is not set to zero, then the device server shall
36       * terminate the command with CHECK CONDITION status with the sense key set to ILLEGAL REQUEST and the
37       * additional sense code set to INVALID FIELD IN CDB.
38       * 
39       * @see #logicalBlockAddress
40       */
41      protected final boolean partialMediumIndicator;
42  
43      public ReadCapacityCdb(final ByteBuffer buffer) {
44          super(buffer);
45          logicalBlockAddress = deserializeLogicalBlockAddress(buffer);
46          partialMediumIndicator = deserializePartialMediumIndicator(buffer);
47          // check constraint
48          if (!partialMediumIndicator && logicalBlockAddress > 0)
49              addIllegalFieldPointer(2);
50      }
