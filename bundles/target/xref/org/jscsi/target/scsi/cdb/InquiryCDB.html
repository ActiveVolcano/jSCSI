12   * This class represents Command Descriptor Blocks for the <code>INQUIRY</code> SCSI command.
13   * 
14   * @author Andreas Ergenzinger
15   */
16  public class InquiryCDB extends CommandDescriptorBlock {
17  
18      /**
19       * An enable vital product data (EVPD) bit set to one specifies that the
20       * device server shall return the vital product data specified by the PAGE
21       * CODE field.
22       * <p>
23       * If the EVPD bit is set to zero, the device server shall return the standard INQUIRY data. If the PAGE
24       * CODE field is not set to zero when the EVPD bit is set to zero, the command shall be terminated with
25       * CHECK CONDITION status, with the sense key set to ILLEGAL REQUEST, and the additional sense code set to
26       * INVALID FIELD IN CDB.
27       */
28      private final boolean enableVitalProductData;
29  
30      /**
31       * The ALLOCATION LENGTH field specifies the maximum number of bytes that an
32       * application client has allocated in the Data-In Buffer. An allocation
33       * length of zero specifies that no data shall be transferred. This
34       * condition shall not be considered as an error. The device server shall
35       * terminate transfers to the Data-In Buffer when the number of bytes
36       * specified by the ALLOCATION LENGTH field have been transferred or when
37       * all available data have been transferred, whichever is less.
38       * <p>
39       * The allocation length is used to limit the maximum amount of variable length data (e.g., mode data, log
40       * data, diagnostic data) returned to an application client. If the information being transferred to the
41       * Data-In Buffer includes fields containing counts of the number of bytes in some or all of the data,
42       * then the contents of these fields shall not be altered to reflect the truncation, if any, that results
43       * from an insufficient ALLOCATION LENGTH value, unless the standard that describes the Data-In Buffer
44       * format states otherwise.
45       * <p>
46       * If the amount of information to be transferred exceeds the maximum value that the ALLOCATION LENGTH
47       * field is capable of specifying, the device server shall transfer no data and terminate the command with
48       * CHECK CONDITION status, with the sense key set to ILLEGAL REQUEST, and the additional sense code set to
49       * INVALID FIELD IN CDB.
50       * <p>
51       * If EVPD is set to zero, the allocation length should be at least five, so that the ADDITIONAL LENGTH
52       * field in the parameter data is returned. If EVPD is set to one, the allocation length should be should
53       * be at least four, so that the PAGE LENGTH field in the parameter data is returned.
54       */
55      private final int allocationLength;
56  
57      /**
58       * When the EVPD bit is set to one, the PAGE CODE field specifies which page
59       * of vital product data information the device server shall return.
60       */
61      private final PageCode pageCode;
62  
63      public InquiryCDB(ByteBuffer buffer) {
64          super(buffer);
65  
66          // EVPD
67          enableVitalProductData = BitManip.getBit(buffer.get(1),// byte
68              0);// bit number
69  
70          // page code
71          pageCode = new PageCode(buffer.get(2));
72  
73          // allocation length
74          allocationLength = ReadWrite.readTwoByteInt(buffer, 3);
75  
76          final VitalProductDataPageName vpdpn = pageCode.getVitalProductDataPageName();
77          if (enableVitalProductData) {
78              if (!SupportedVpdPages.vpdPageCodeSupported(vpdpn))
79                  addIllegalFieldPointer(2);// page code not supported
80          } else {
81              /*
82               * If the PAGE CODE field is not set to zero when the EVPD bit is
83               * set to zero, the command shall be terminated with CHECK CONDITION
84               * status, with the sense key set to ILLEGAL REQUEST, and the
85               * additional sense code set to INVALID FIELD IN CDB.
86               */
87              if (vpdpn != VitalProductDataPageName.SUPPORTED_VPD_PAGES)
88                  addIllegalFieldPointer(2);// value should be 0x00
89          }
90      }
