57      public void serialize(ByteBuffer byteBuffer, int index) {
58  
59          // *** byte 0 ****
60          /*
61           * Peripheral qualifier (3 most significant bits of byte 0):
62           * 
63           * 000b
64           * 
65           * A peripheral device having the specified peripheral device type is
66           * connected to this logical unit. If the device server is unable to
67           * determine whether or not a peripheral device is connected, it also
68           * shall use this peripheral qualifier. This peripheral qualifier does
69           * not mean that the peripheral device connected to the logical unit is
70           * ready for access.
71           * 
72           * Peripheral Device Type (5 least significant bits of byte 0):
73           * 
74           * 00000b
75           * 
76           * direct access block device
77           * 
78           * Therefore no need to change anything.
79           */
80          byteBuffer.position(index);
81          byteBuffer.put((byte)0);
82  
83          // *** byte 1 ***
84          /*
85           * RMB bit (bit 7):
86           * 
87           * 0
88           * 
89           * A removable medium (RMB) bit set to zero indicates that the medium is
90           * not removable.
91           * 
92           * (remaining bits are RESERVED)
93           */
94          byteBuffer.put((byte)0);
95  
96          // *** byte 2 ***
97          /*
98           * version:
99           * 
100          * 0x05
101          * 
102          * The device complies to the SPC3 and respective ANSI standard
103          */
104         byteBuffer.put((byte)0x05);
105 
106         // *** byte 3 - flags ***
107         /*
108          * (bits 7 and 6 are obsolete)
109          * 
110          * NORMACA (bit 5):
111          * 
112          * 0
113          * 
114          * A NORMACA (normal auto contingency allegiance) bit set to zero
115          * indicates that the device server does not support a NACA bit set to
116          * one and does not support the ACA task attribute.
117          * 
118          * HISUP (bit 4):
119          * 
120          * 0
121          * 
122          * A hierarchical support (HISUP) bit set to zero indicates the SCSI
123          * target device does not use the hierarchical addressing model to
124          * assign LUNs to logical units.
125          * 
126          * RESPONSE DATA FORMAT (bits 3-0):
127          * 
128          * 0010b
129          * 
130          * RESPONSE DATA FORMAT: SPC3
131          */
132         byteBuffer.put((byte)2);// 0000 0010b
133 
134         // *** byte 4 ***
135         /*
136          * n - 4 = 35 - 4 = 31
137          * 
138          * The ADDITIONAL LENGTH field indicates the length in bytes of the
139          * remaining standard INQUIRY data.
140          */
141         byteBuffer.put((byte)31);
142 
143         // *** byte 5 ***
144         /*
145          * SSCS (bit 7):
146          * 
147          * 0
148          * 
149          * An SCC Supported bit set to zero indicates that the SCSI target
150          * device does not contain an embedded storage array controller
151          * component.
152          * 
153          * ACC (bit 6):
154          * 
155          * 0
156          * 
157          * An ACC bit set to zero indicates that no access controls coordinator
158          * may be addressed through this logical unit.
159          * 
160          * TGPS field (bits 5 and 4):
161          * 
162          * 00b
163          * 
164          * The contents of the target port group support (TPGS) indicate the
165          * support for asymmetric logical unit access. The SCSI target device
166          * does not support asymmetric logical unit access or supports a form of
167          * asymmetric access that is vendor specific. Neither the REPORT TARGET
168          * GROUPS nor the SET TARGET GROUPS commands is supported.
169          * 
170          * 3PC (bit 3):
171          * 
172          * 0
173          * 
174          * A Third-Party Copy (3PC) bit set to zero indicates that the SCSI
175          * target device does not support third-party copy commands such as the
176          * EXTENDED COPY command.
177          * 
178          * (bits 2 and 1 are RESERVED)
179          * 
180          * PROTECT (bit 0):
181          * 
182          * 0
183          * 
184          * A PROTECT bit set to zero indicates that the logical unit does not
185          * support protection information.
186          */
187         byteBuffer.put((byte)0);
188 
189         // *** byte 6 ***
190         /*
191          * BQUE (bit 7):
192          * 
193          * The CMDQUE bit and BQUE bit indicate whether the logical unit
194          * supports the full task management model.
195          * 
196          * 1 (and 0)
197          * 
198          * Basic task management model supported.
199          * 
200          * ENCSERV (bit 6):
201          * 
202          * 0
203          * 
204          * An ENCSERV bit set to zero indicates that the SCSI target device does
205          * not contain an embedded enclosure services component.
206          * 
207          * (bit 5 is VENDOR SPECIFIC i.e. 0)
208          * 
209          * MULTIP (bit 4):
210          * 
211          * 0
212          * 
213          * A Multi Port (MULTIP) bit set to zero indicates that this is not a
214          * multi- port (two or more ports) SCSI target device and therefore does
215          * not conform to the SCSI multi-port device requirements found in the
216          * applicable standards (e.g., SAM-3, a SCSI transport protocol standard
217          * and possibly provisions of a command standard).
218          * 
219          * MCHNGR (bit 3):
220          * 
221          * 0
222          * 
223          * The MCHNGR bit is valid only when the RMB bit is equal to one. A
224          * MCHNGR bit set to zero indicates that the SCSI target device does not
225          * support commands to control an attached media changer.
226          * 
227          * (bits 2 and 1 are RESERVED)
228          * 
229          * ADDR16 (bit 0):
230          * 
231          * 0
232          * 
233          * RESERVED since the SCSI Parallel Interface transport protocol is not
234          * used.
235          */
236         byteBuffer.put((byte)128);// 1000 0000b
237 
238         // *** byte 7 ***
239         /*
240          * (bits 7, 6, and 2 are OBSOLETE, bit 0 is VENDOR SPECIFIC)
241          * 
242          * WBUS and SYNC (bits 5 and 4):
243          * 
244          * 00b
245          * 
246          * RESERVED since the SCSI Parallel Interface transport protocol is not
247          * used.
248          * 
249          * 
250          * LINKED (bit 3):
251          * 
252          * 0
253          * 
254          * A LINKED bit set to zero indicates the device server does not support
255          * linked commands.
256          * 
257          * CMDQUE (bit 1):
258          * 
259          * 0
260          * 
261          * The CMDQUE bit and BQUE bit indicate whether the logical unit
262          * supports the full task management model.
263          * 
264          * (see BQUE ,byte 6)
265          */
266         byteBuffer.put((byte)0);
267 
268         // *** bytes 8 to 15 ***
269         /*
270          * T10 VENDOR IDENTIFICATION:
271          * 
272          * The T10 VENDOR IDENTIFICATION field contains eight bytes of
273          * left-aligned ASCII data identifying the vendor of the product.
274          * 
275          * The T10 vendor identification shall be one assigned by INCITS, but
276          * obviously that is not the case here. disyUKon
277          */
278         putString(byteBuffer, VENDOR_ID, index + VENDOR_ID_FIELD_POSITION, VENDOR_ID_FIELD_LENGTH);
279 
280         // *** bytes 16 to 31
281         /*
282          * PRODUCT IDENTIFICATION:
283          * 
284          * The PRODUCT IDENTIFICATION field contains sixteen bytes of
285          * left-aligned ASCII data defined by the vendor.
286          */
287         putString(byteBuffer, PRODUCT_ID, index + PRODUCT_ID_FIELD_POSITION, PRODUCT_ID_FIELD_LENGTH);
288 
289         // *** bytes 32 to 35 ***
290         /*
291          * PRODUCT REVISION LEVEL:
292          * 
293          * The PRODUCT REVISION LEVEL field contains four bytes of left-aligned
294          * ASCII data defined by the vendor.
295          */
296         putString(byteBuffer, PRODUCT_REVISION_LEVEL, index + PRODUCT_REVISION_LEVEL_FIELD_POSITION,
297             PRODUCT_REVISION_LEVEL_FIELD_LENGTH);
298     }

300     /**
301      * Puts up to <i>fieldLength</i> of the passed {@link String} into a {@link ByteBuffer} and fills the
302      * remaining bytes with zeros.
303      * 
304      * @param byteBuffer
305      *            where the {@link String}'s characters will be copied
306      * @param string
307      *            contains the characters to copy
308      * @param position
309      *            where the first character of the {@link String} will be put
310      * @param fieldLength
311      *            the number of bytes in the {@link ByteBuffer} that will be
312      *            overwritten
313      */
314     private void putString(final ByteBuffer byteBuffer, final String string, final int position,
315         final int fieldLength) {
316         // set position
317         byteBuffer.position(position);
318         // put string characters
319         final int stringLength = Math.min(string.length(), fieldLength);
320         for (int i = 0; i < stringLength; ++i)
321             byteBuffer.put((byte)string.charAt(i));
322         // clear remaining remaining bytes
323         for (int i = stringLength; i < fieldLength; ++i)
324             byteBuffer.put((byte)0);
325     }
