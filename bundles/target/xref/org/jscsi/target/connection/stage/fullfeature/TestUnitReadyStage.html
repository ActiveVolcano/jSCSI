22   * @author Andreas Ergenzinger
23   */
24  public class TestUnitReadyStage extends TargetFullFeatureStage {
25  
26      public TestUnitReadyStage(TargetFullFeaturePhase targetFullFeaturePhase) {
27          super(targetFullFeaturePhase);
28      }
29  
30      @Override
31      public void execute(ProtocolDataUnit pdu) throws IOException, InterruptedException,
32          InternetSCSIException, DigestException, SettingsException {
33  
34          final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
35          final SCSICommandParser parser = (SCSICommandParser)bhs.getParser();
36  
37          ProtocolDataUnit responsePdu;// the response PDU
38  
39          // get command details in CDB
40          final TestUnitReadyCdb cdb = new TestUnitReadyCdb(parser.getCDB());
41          final FieldPointerSenseKeySpecificData[] illegalFieldPointers = cdb.getIllegalFieldPointers();
42  
43          if (illegalFieldPointers != null) {
44              // an illegal request has been made
45  
46              responsePdu = createFixedFormatErrorPdu(illegalFieldPointers,// senseKeySpecificData,
47                  bhs.getInitiatorTaskTag(),// initiatorTaskTag,
48                  parser.getExpectedDataTransferLength());// expDataTransferLength
49  
50          } else {
51              // PDU is okay
52              // carry out command
53              // the logical unit is always ready
54              responsePdu = TargetPduFactory.createSCSIResponsePdu(false,// bidirectionalReadResidualOverflow
55                  false,// bidirectionalReadResidualUnderflow
56                  false,// residualOverflow
57                  false,// residualUnderflow,
58                  SCSIResponseParser.ServiceResponse.COMMAND_COMPLETED_AT_TARGET,// response,
59                  SCSIStatus.GOOD,// status,
60                  bhs.getInitiatorTaskTag(),// initiatorTaskTag,
61                  0,// snackTag
62                  0,// expectedDataSequenceNumber
63                  0,// bidirectionalReadResidualCount
64                  0,// residualCount
65                  ScsiResponseDataSegment.EMPTY_DATA_SEGMENT);// data
66                                                              // segment
67          }
68  
69          // send response
70          connection.sendPdu(responsePdu);
71      }
