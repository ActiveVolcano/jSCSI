25  /**
26   * A stage for processing <code>MODE SENSE (6)</code> SCSI commands.
27   * 
28   * @author Andreas Ergenzinger
29   */
30  public final class ModeSenseStage extends TargetFullFeatureStage {
31  
32      private static final Logger LOGGER = Logger.getLogger(ModeSenseStage.class);
33  
34      public ModeSenseStage(TargetFullFeaturePhase targetFullFeaturePhase) {
35          super(targetFullFeaturePhase);
36      }
37  
38      @Override
39      public void execute(final ProtocolDataUnit pdu) throws IOException, InterruptedException,
40          InternetSCSIException, DigestException, SettingsException {
41  
42          final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
43          final SCSICommandParser parser = (SCSICommandParser)bhs.getParser();
44          final ModeSense6Cdb cdb = new ModeSense6Cdb(parser.getCDB());
45  
46          if (LOGGER.isDebugEnabled()) {
47              LOGGER.debug(cdb.getDisableBlockDescriptors());
48              LOGGER.debug(cdb.getPageControl());
49              LOGGER.debug(cdb.getPageCode());
50              LOGGER.debug(cdb.getSubpageCode());
51              LOGGER.debug("cdb.getAllocationLength() = " + cdb.getAllocationLength());
52              LOGGER.debug(cdb.getModePage());
53          }
54  
55          // final PageControl pageControl = cdb.getPageControl();//see 8 lines
56          // below
57          final ModePageCode modePageCode = cdb.getModePage();
58  
59          // ModeParameterList and ModeParameterListBuilder common to
60          // all supported ModePage requests
61  
62          ModePage[] modePages = null;
63          if (modePageCode == ModePageCode.INFORMATIONAL_EXCEPTIONS_CONTROL_MODE_PAGE) {
64              // TODO this should to be made dynamic wrt. cdb.getPageControl();
65  
66              modePages = new ModePage[] {
67                  getInformationExceptionsControlModePage()
68              };
69  
70          } else if (modePageCode == ModePageCode.CACHING_MODE_PAGE) {
71  
72              modePages = new ModePage[] {
73                  getCachingModePage()
74              };
75  
76          } else if (modePageCode == ModePageCode.RETURN_ALL_MODE_PAGES_ONLY) {
77  
78              modePages = new ModePage[] {
79                  getInformationExceptionsControlModePage(), getCachingModePage()
80              };
81  
82          }// else modeParameterList stays null
83  
84          // create and send response PDU
85          if (modePages != null) {
86  
87              // create ModeParameterList
88              final ModeParameterListBuilder builder =
89                  new ModeParameterListBuilder(HeaderType.MODE_PARAMETER_HEADER_6);
90              builder.setLogicalBlockDescriptors(new ShortLogicalBlockDescriptor(session.getStorageModule()
91                  .getSizeInBlocks(),// numberOfLogicalBlocks
92                  VIRTUAL_BLOCK_SIZE));// logicalBlockLength
93              builder.setModePages(modePages);
94              ModeParameterList modeParameterList = ModeParameterList.build(builder);
95  
96              // send response
97              sendResponse(bhs.getInitiatorTaskTag(),// initiatorTaskTag,
98                  parser.getExpectedDataTransferLength(),// expectedDataTransferLength,
99                  modeParameterList);// responseData
100 
101         } else {
102             /*
103              * The initiator has requested a mode sense page which the jSCSI
104              * Target cannot provide.
105              * 
106              * This could be answered with an Illegal field in CDB message but,
107              * there is no good way to identify the exact field at fault.
108              */
109             throw new InternetSCSIException();
110         }
111 
112     }
