34      @Override
35      public void execute (ProtocolDataUnit pdu) throws IOException , InterruptedException , InternetSCSIException , DigestException , SettingsException {
36  
37          final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
38          final SCSICommandParser parser = (SCSICommandParser) bhs.getParser();
39  
40          ProtocolDataUnit responsePdu = null;// the response PDU
41  
42          // get command details in CDB
43          final ReportLunsCDB cdb = new ReportLunsCDB(parser.getCDB());
44          final FieldPointerSenseKeySpecificData[] illegalFieldPointers = cdb.getIllegalFieldPointers();
45  
46          if (illegalFieldPointers != null) {
47              // an illegal request has been made
48              responsePdu = createFixedFormatErrorPdu(illegalFieldPointers,// senseKeySpecificData
49                      bhs.getInitiatorTaskTag(),// initiatorTaskTag
50                      parser.getExpectedDataTransferLength());// expectedDataTransferLength
51  
52              // send response
53              connection.sendPdu(responsePdu);
54  
55          } else {
56              // PDU is okay
57              // carry out command
58              final SelectReport selectReport = cdb.getSelectReport();
59              LOGGER.debug("selectReport = " + selectReport);
60  
61              // there are only well known LUNs
62              ReportLunsParameterData reportLunsParameterData;
63  
64              // TODO the switch isn't really needed right now, but maybe in
65              // future implementations
66              switch (selectReport) {
67                  case SELECTED_ADDRESSING_METHODS :
68                  case WELL_KNOWN_LUNS_ONLY :
69                  case ALL :
70                      reportLunsParameterData = new ReportLunsParameterData(session.getTargetServer().getConfig().getLogicalUnitNumber());
71                      break;
72                  default :
73                      throw new InternetSCSIException();
74                      /*
75                       * Unreachable, this case has already been checked in the ReportLunsCDB constructor
76                       */
77              }
78  
79              // send response
80              sendResponse(bhs.getInitiatorTaskTag(),// initiatorTaskTag
81                      parser.getExpectedDataTransferLength(),// expectedDataTransferLength
82                      reportLunsParameterData);// responseData
83  
84          }
85      }
