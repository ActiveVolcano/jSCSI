25  /**
26   * A stage for processing <code>READ (6)</code> and <code>READ (10)</code> SCSI
27   * commands.
28   * 
29   * @author Andreas Ergenzinger
30   */
31  public class ReadStage extends ReadOrWriteStage {
32  
33      private static final Logger LOGGER = LoggerFactory.getLogger(ReadStage.class);
34  
35      public ReadStage(final TargetFullFeaturePhase targetFullFeaturePhase) {
36          super(targetFullFeaturePhase);
37      }
38  
39      @Override
40      public void execute(ProtocolDataUnit pdu) throws IOException, InterruptedException,
41          InternetSCSIException, SettingsException {
42  
43          // get relevant variables ...
44          // ... from settings
45          final boolean immediateData = settings.getImmediateData();
46  
47          if (LOGGER.isDebugEnabled()) {
48              LOGGER.debug("immediateData = " + immediateData);
49              LOGGER.debug("maxRecvDataSegmentLength = " + settings.getMaxRecvDataSegmentLength());
50          }
51  
52          // ... and from the PDU
53          BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
54          SCSICommandParser parser = (SCSICommandParser)bhs.getParser();
55          final int initiatorTaskTag = bhs.getInitiatorTaskTag();
56  
57          // get the Read(6) or Read(10) CDB
58          ReadCdb cdb;
59          final ScsiOperationCode scsiOpCode = ScsiOperationCode.valueOf(parser.getCDB().get(0));
60          if (scsiOpCode == ScsiOperationCode.READ_10)// most likely option first
61              cdb = new Read10Cdb(parser.getCDB());
62          else if (scsiOpCode == ScsiOperationCode.READ_6)
63              cdb = new Read6Cdb(parser.getCDB());
64          else {
65              // anything else wouldn't be good (programmer error)
66              // close connection
67              throw new InternetSCSIException("wrong SCSI Operation Code " + scsiOpCode + " in ReadStage");
68          }
69  
70          // check if requested blocks are out of bounds
71          checkOverAndUnderflow(cdb);
72  
73          // check illegal field pointers
74          if (cdb.getIllegalFieldPointers() != null) {
75              // the command must fail
76  
77              LOGGER.error("illegal field in Read CDB");
78  
79              // create and send error PDU and leave stage
80              final ProtocolDataUnit responsePdu = createFixedFormatErrorPdu(cdb.getIllegalFieldPointers(),// senseKeySpecificData
81                  initiatorTaskTag, parser.getExpectedDataTransferLength());
82              connection.sendPdu(responsePdu);
83              return;
84          }
85  
86          final int totalTransferLength = VIRTUAL_BLOCK_SIZE * cdb.getTransferLength();
87          final long storageOffset = VIRTUAL_BLOCK_SIZE * cdb.getLogicalBlockAddress();
88  
89          if (LOGGER.isDebugEnabled()) {
90              LOGGER.debug("cdb.getLogicalBlockAddress() = " + cdb.getLogicalBlockAddress());
91              LOGGER.debug("blockSize = " + VIRTUAL_BLOCK_SIZE);
92              LOGGER.debug("totalTransferLength = " + totalTransferLength);
93              LOGGER.debug("expectedDataSegmentLength = " + parser.getExpectedDataTransferLength());
94          }
95  
96          // *** start sending ***
97          // initialize counters and data segment buffer
98          int bytesSent = 0;
99          int dataSequenceNumber = 0;
100         byte[] dataSegmentArray = null;
101         ByteBuffer dataSegment = null;
102         ProtocolDataUnit responsePdu;
103 
104         // *** send up to last but one Data-In PDU ***
105         // (with DataSegmentSize == MaxRecvDataSegmentLength)
106 
107         if (bytesSent < totalTransferLength - settings.getMaxRecvDataSegmentLength()) {
108             /*
109              * Initialize dataSegmentArray and dataSegment with
110              * MaxRecvDataSegmentLength bytes.
111              */
112             dataSegmentArray = connection.getDataInArray(settings.getMaxRecvDataSegmentLength());
113             dataSegment = ByteBuffer.wrap(dataSegmentArray);
114         }
115 
116         while (bytesSent < totalTransferLength - settings.getMaxRecvDataSegmentLength()) {
117 
118             // get data and prepare data segment
119             session.getStorageModule().read(dataSegmentArray, storageOffset + bytesSent);
120 
121             // create and send PDU
122             responsePdu = TargetPduFactory.createDataInPdu(false,// finalFlag,
123                                                                  // not the last
124                                                                  // PDU with
125                                                                  // data payload
126                                                                  // in the
127                                                                  // sequence
128                 false,// acknowledgeFlag, ErrorRecoveryLevel == 0, so we
129                       // never do that
130                 false,// residualOverflowFlag
131                 false,// residualUnderflowFlag
132                 false,// statusFlag
133                 SCSIStatus.GOOD,// status, actually reserved i.e. 0x0
134                 0L,// logicalUnitNumber, reserved
135                 initiatorTaskTag, 0xffffffff,// targetTransferTag
136                 dataSequenceNumber,// dataSequenceNumber
137                 bytesSent,// bufferOffset
138                 0,// residualCount
139                 dataSegment);
140 
141             connection.sendPdu(responsePdu);
142 
143             // increment counters
144             ++dataSequenceNumber;
145             bytesSent += settings.getMaxRecvDataSegmentLength();
146         }
147 
148         /*
149          * If ImmediateData=Yes has been negotiated, then a phase collapse has
150          * to take place, i.e. the status is sent in the last Data-In PDU.
151          * Otherwise a separate SCSI Response PDU must follow.
152          */
153 
154         // *** send last Data-In PDU ***
155 
156         // get data and prepare data segment
157         final int bytesRemaining = totalTransferLength - bytesSent;
158         dataSegmentArray = connection.getDataInArray(bytesRemaining);
159         session.getStorageModule().read(dataSegmentArray, storageOffset + bytesSent);
160         dataSegment = ByteBuffer.wrap(dataSegmentArray);
161 
162         // create and send PDU (with or without status)
163         responsePdu = TargetPduFactory.createDataInPdu(true,// finalFlag, last
164                                                             // PDU in the
165                                                             // sequence with
166                                                             // data payload
167             false,// acknowledgeFlag, ErrorRecoveryLevel == 0, so we never
168                   // do that
169             false,// residualOverflowFlag
170             false,// residualUnderflowFlag
171             immediateData,// statusFlag
172             SCSIStatus.GOOD,// status, or not (reserved if no status)
173             0L,// logicalUnitNumber, reserved
174             initiatorTaskTag, 0xffffffff,// targetTransferTag
175             dataSequenceNumber,// dataSequenceNumber
176             bytesSent,// bufferOffset
177             0,// residualCount
178             dataSegment);
179 
180         LOGGER.debug("sending last Data-In PDU");
181         connection.sendPdu(responsePdu);
182 
183         // send SCSI Response PDU?
184         if (!immediateData) {
185 
186             responsePdu = TargetPduFactory.createSCSIResponsePdu(false,// bidirectionalReadResidualOverflow
187                 false,// bidirectionalReadResidualUnderflow
188                 false,// residualOverflow
189                 false,// residualUnderflow
190                 ServiceResponse.COMMAND_COMPLETED_AT_TARGET,// response
191                 SCSIStatus.GOOD,// status
192                 initiatorTaskTag,// initiatorTaskTag
193                 0,// snackTag, reserved
194                 0,// expectedDataSequenceNumber, reserved
195                 0,// bidirectionalReadResidualCount
196                 0,// residualCount
197                 ScsiResponseDataSegment.EMPTY_DATA_SEGMENT);// empty
198                                                             // ScsiResponseDataSegment
199 
200             LOGGER.debug("sending SCSI Response PDU");
201             connection.sendPdu(responsePdu);
202         }
203 
204     }
