14  /**
15   * A stage for processing logout requests.
16   * <p>
17   * Since <code>MaxConnections</code> is currently limited to <code>1</code>, all logout requests will be
18   * treated as requests to close the session.
19   * 
20   * @author Andreas Ergenzinger
21   */
22  public final class LogoutStage extends TargetFullFeatureStage {
23  
24      public LogoutStage(TargetFullFeaturePhase targetFullFeaturePhase) {
25          super(targetFullFeaturePhase);
26      }
27  
28      @Override
29      public void execute(ProtocolDataUnit pdu) throws IOException, InterruptedException,
30          InternetSCSIException, DigestException, SettingsException {
31  
32          final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
33          final int initiatorTaskTag = bhs.getInitiatorTaskTag();
34  
35          final ProtocolDataUnit responsePDU =
36              TargetPduFactory.createLogoutResponsePdu(LogoutResponse.CONNECTION_CLOSED_SUCCESSFULLY,
37                  initiatorTaskTag, (short)settings.getDefaultTime2Wait(),// time2Wait
38                  (short)settings.getDefaultTime2Retain());// time2Retain
39  
40          connection.sendPdu(responsePDU);
41      }

43  }
