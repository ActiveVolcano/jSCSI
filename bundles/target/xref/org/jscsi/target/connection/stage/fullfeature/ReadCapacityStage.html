27  
28      private static final Logger LOGGER = LoggerFactory.getLogger(ReadCapacityStage.class);
29  
30      public ReadCapacityStage(final TargetFullFeaturePhase targetFullFeaturePhase) {
31          super(targetFullFeaturePhase);
32      }
33  
34      @Override
35      public void execute(ProtocolDataUnit pdu) throws IOException, InterruptedException,
36          InternetSCSIException, DigestException, SettingsException {
37  
38          // find out the type of READ CAPACITY command ((10) or (16))
39          final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
40          final SCSICommandParser parser = (SCSICommandParser)bhs.getParser();
41          final ScsiOperationCode opCode = ScsiOperationCode.valueOf(parser.getCDB().get(0));
42          ReadCapacityCdb cdb;
43          if (opCode == ScsiOperationCode.READ_CAPACITY_10)
44              cdb = new ReadCapacity10Cdb(parser.getCDB());
45          else if (opCode == ScsiOperationCode.READ_CAPACITY_16)
46              cdb = new ReadCapacity16Cdb(parser.getCDB());
47          else {
48              // programmer error, we should not be here, close the connection
49              throw new InternetSCSIException("wrong SCSI Operation Code " + opCode + " in ReadCapacityStage");
50          }
51  
52          /*
53           * Everything is fine, carry on.
54           * 
55           * The PMI bit of the command descriptor block is ignored, since there
56           * is no way to know if "substantial vendor specific delay in data
57           * transfer may be encountered" after the address in the LOGICAL BLOCK
58           * ADDRESS field. Therefore we always try to return the whole length of
59           * the storage medium.
60           */
61  
62          // make sure that the LOGICAL BLOCK ADDRESS field is valid and send
63          // appropriate response
64          if (session.getStorageModule().checkBounds(cdb.getLogicalBlockAddress(), 0) != 0) {
65              // invalid, log error, send error PDU, and return
66              LOGGER.error("encountered " + cdb.getClass() + " in ReadCapacityStage with "
67                  + "LOGICAL BLOCK ADDRESS = " + cdb.getLogicalBlockAddress());
68  
69              final FieldPointerSenseKeySpecificData fp = new FieldPointerSenseKeySpecificData(true,// senseKeySpecificDataValid
70                  true,// commandData (i.e. invalid field in CDB)
71                  false,// bitPointerValid
72                  0,// bitPointer, reserved since invalid
73                  0);// fieldPointer to the SCSI OpCode field
74              final FieldPointerSenseKeySpecificData[] fpArray = new FieldPointerSenseKeySpecificData[] {
75                  fp
76              };
77              final ProtocolDataUnit responsePdu = createFixedFormatErrorPdu(fpArray,// senseKeySpecificData
78                  AdditionalSenseCodeAndQualifier.LOGICAL_BLOCK_ADDRESS_OUT_OF_RANGE,// additionalSenseCodeAndQualifier
79                  bhs.getInitiatorTaskTag(),// initiatorTaskTag
80                  parser.getExpectedDataTransferLength());// expectedDataTransferLength
81              connection.sendPdu(responsePdu);
82              return;
83          } else {
84              // send PDU with requested READ CAPACITY parameter data
85              ReadCapacityParameterData parameterData;
86              if (cdb instanceof ReadCapacity10Cdb)
87                  parameterData = new ReadCapacity10ParameterData(session.getStorageModule().getSizeInBlocks(),// returnedLogicalBlockAddress
88                      VIRTUAL_BLOCK_SIZE);// logicalBlockLengthInBytes
89              else
90                  parameterData = new ReadCapacity16ParameterData(session.getStorageModule().getSizeInBlocks(),// returnedLogicalBlockAddress
91                      VIRTUAL_BLOCK_SIZE);// logicalBlockLengthInBytes
92  
93              sendResponse(bhs.getInitiatorTaskTag(),// initiatorTaskTag,
94                  parser.getExpectedDataTransferLength(),// expectedDataTransferLength,
95                  parameterData);// responseData
96          }
97      }
