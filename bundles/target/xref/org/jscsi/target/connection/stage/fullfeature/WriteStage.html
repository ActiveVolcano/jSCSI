29  /**
30   * A stage for processing <code>WRITE (6)</code> and <code>WRITE (10)</code> SCSI commands.
31   * 
32   * @author Andreas Ergenzinger
33   */
34  public final class WriteStage extends ReadOrWriteStage {
35  
36      private static final Logger LOGGER = LoggerFactory.getLogger(WriteStage.class);
37  
38      /**
39       * The <code>DataSN</code> value the next Data-Out PDU must carry.
40       */
41      private int expectedDataSequenceNumber = 0;
42  
43      public WriteStage(TargetFullFeaturePhase targetFullFeaturePhase) {
44          super(targetFullFeaturePhase);
45      }
46  
47      /**
48       * Is used for checking if the PDUs received in a Data-Out sequence actually
49       * are Data-Out PDU and if the PDUs have been received in order.
50       * 
51       * @param parser
52       *            the {@link AbstractMessageParser} subclass instance retrieved
53       *            from the {@link ProtocolDataUnit}'s {@link BasicHeaderSegment}
54       * @throws InternetSCSIException
55       *             if an unexpected PDU has been received
56       */
57      private void checkDataOutParser(final AbstractMessageParser parser) throws InternetSCSIException {
58          if (parser instanceof DataOutParser) {
59              final DataOutParser p = (DataOutParser)parser;
60              if (p.getDataSequenceNumber() == expectedDataSequenceNumber++)
61                  return;
62          }
63          throw new InternetSCSIException("received erroneous PDU in data-out sequence");
64      }
65  
66      @Override
67      public void execute(ProtocolDataUnit pdu) throws IOException, DigestException, InterruptedException,
68          InternetSCSIException, SettingsException {
69  
70          if (LOGGER.isDebugEnabled())
71              LOGGER.debug("Entering WRITE STAGE");
72  
73          // get relevant values from settings
74          final boolean immediateData = settings.getImmediateData();
75          final boolean initialR2T = settings.getInitialR2T();
76          final int firstBurstLength = settings.getFirstBurstLength();
77          final int maxBurstLength = settings.getMaxBurstLength();
78  
79          if (LOGGER.isDebugEnabled()) {
80              LOGGER.debug("immediateData = " + immediateData);
81              LOGGER.debug("initialR2T = " + initialR2T);
82          }
83  
84          // get relevant values from PDU/CDB
85          BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
86          SCSICommandParser parser = (SCSICommandParser)bhs.getParser();
87          final int initiatorTaskTag = bhs.getInitiatorTaskTag();
88          WriteCdb cdb;
89          final ScsiOperationCode scsiOpCode = ScsiOperationCode.valueOf(parser.getCDB().get(0));
90          if (scsiOpCode == ScsiOperationCode.WRITE_10)
91              cdb = new Write10Cdb(parser.getCDB());
92          else if (scsiOpCode == ScsiOperationCode.WRITE_6)
93              cdb = new Write6Cdb(parser.getCDB());
94          else {
95              // anything else wouldn't be good (programmer error)
96              // close connection
97              throw new InternetSCSIException("wrong SCSI Operation Code " + scsiOpCode + " in WriteStage");
98          }
99          final int transferLength = cdb.getTransferLength();
100         final long logicalBlockAddress = cdb.getLogicalBlockAddress();
101 
102         // transform to from block units to byte units
103         final int transferLengthInBytes = transferLength * VIRTUAL_BLOCK_SIZE;
104         long storageIndex = logicalBlockAddress * VIRTUAL_BLOCK_SIZE;
105 
106         // check if requested blocks are out of bounds
107         // (might add FPSKSD to the CDB's list to be detected in the next step)
108         checkOverAndUnderflow(cdb);
109 
110         if (cdb.getIllegalFieldPointers() != null) {
111             /*
112              * CDB is invalid, inform initiator by closing the connection.
113              * 
114              * Sending an error status SCSI Response PDU will not work reliably,
115              * since the initiator may not be expecting a response so soon.
116              * Also, if the WriteStage is simply left early (without closing the
117              * connection), the initiator may send additional unsolicited
118              * Data-Out PDUs, which the jSCSI Target is currently unable to
119              * ignore or process properly.
120              */
121             LOGGER.error("illegal field in Write CDB");
122             LOGGER.error("CDB:\n" + Debug.byteBufferToString(parser.getCDB()));
123             throw new InternetSCSIException();// leads to connection closing
124         }
125 
126         // *** start receiving data (or process what has already been sent) ***
127         int bytesReceived = 0;
128 
129         // *** receive immediate data ***
130         if (immediateData && bhs.getDataSegmentLength() > 0) {
131             final byte[] immediateDataArray = pdu.getDataSegment().array();
132 
133             session.getStorageModule().write(immediateDataArray, 0, immediateDataArray.length, storageIndex);
134             bytesReceived = immediateDataArray.length;
135 
136             if (LOGGER.isDebugEnabled())
137                 LOGGER.debug("wrote " + immediateDataArray.length + "bytes as immediate data");
138         }
139 
140         // *** receive unsolicited data ***
141         if (!initialR2T && !bhs.isFinalFlag()) {
142 
143             if (LOGGER.isDebugEnabled())
144                 LOGGER.debug("receiving unsolicited data");
145 
146             boolean firstBurstOver = false;
147             while (!firstBurstOver && bytesReceived <= firstBurstLength) {
148 
149                 // receive and check PDU
150                 pdu = connection.receivePdu();
151                 bhs = pdu.getBasicHeaderSegment();
152 
153                 checkDataOutParser(bhs.getParser());
154 
155                 final DataOutParser dataOutParser = (DataOutParser)bhs.getParser();
156 
157                 session.getStorageModule().write(pdu.getDataSegment().array(), 0,
158                     pdu.getDataSegment().array().length, storageIndex + dataOutParser.getBufferOffset());
159                 ;
160                 bytesReceived += bhs.getDataSegmentLength();
161 
162                 if (bhs.isFinalFlag())
163                     firstBurstOver = true;
164             }
165         }
166 
167         // *** receive solicited data ***
168         if (bytesReceived < transferLengthInBytes) {
169             if (LOGGER.isDebugEnabled())
170                 LOGGER.debug(bytesReceived + "<" + transferLengthInBytes);
171 
172             int readyToTransferSequenceNumber = 0;
173             int desiredDataTransferLength;
174 
175             while (bytesReceived < transferLengthInBytes) {
176 
177                 desiredDataTransferLength = Math.min(maxBurstLength, transferLengthInBytes - bytesReceived);
178 
179                 // send R2T
180                 pdu = TargetPduFactory.createReadyToTransferPdu(0,// logicalUnitNumber
181                     initiatorTaskTag, TargetServer.getNextTargetTransferTag(),// targetTransferTag
182                     readyToTransferSequenceNumber++, bytesReceived,// bufferOffset
183                     desiredDataTransferLength);
184 
185                 connection.sendPdu(pdu);
186 
187                 // receive DataOut PDUs
188                 expectedDataSequenceNumber = 0;// reset sequence counter//FIXME
189                                                // fix in jSCSI Initiator
190                 boolean solicitedDataCycleOver = false;
191                 int bytesReceivedThisCycle = 0;
192                 while (!solicitedDataCycleOver) {
193 
194                     // receive and check PDU
195                     pdu = connection.receivePdu();
196                     bhs = pdu.getBasicHeaderSegment();
197                     checkDataOutParser(bhs.getParser());
198 
199                     final DataOutParser dataOutParser = (DataOutParser)bhs.getParser();
200 
201                     session.getStorageModule().write(pdu.getDataSegment().array(), 0,
202                         pdu.getDataSegment().array().length, storageIndex + dataOutParser.getBufferOffset());
203 
204                     bytesReceivedThisCycle += bhs.getDataSegmentLength();
205 
206                     /*
207                      * Checking the final flag should be enough, but is not,
208                      * when dealing with the jSCSI Initiator. This is also one
209                      * of the reasons, why the contents of this while loop,
210                      * though very similar to what is happening during the
211                      * receiving of the unsolicited data PDU sequence, has not
212                      * been put into a dedicated method.
213                      */
214                     if (bhs.isFinalFlag() || bytesReceivedThisCycle >= desiredDataTransferLength)
215                         solicitedDataCycleOver = true;
216                 }
217                 bytesReceived += bytesReceivedThisCycle;
218             }
219         }
220 
221         /* send SCSI Response PDU */
222         pdu = TargetPduFactory.createSCSIResponsePdu(false,// bidirectionalReadResidualOverflow
223             false,// bidirectionalReadResidualUnderflow
224             false,// residualOverflow
225             false,// residualUnderflow
226             SCSIResponseParser.ServiceResponse.COMMAND_COMPLETED_AT_TARGET,// response
227             SCSIStatus.GOOD,// status
228             initiatorTaskTag, 0,// snackTag
229             0,// (ExpDataSN or) Reserved
230             0,// bidirectionalReadResidualCount
231             0,// residualCount
232             ScsiResponseDataSegment.EMPTY_DATA_SEGMENT);// dataSegment
233 
234         connection.sendPdu(pdu);
235     }
