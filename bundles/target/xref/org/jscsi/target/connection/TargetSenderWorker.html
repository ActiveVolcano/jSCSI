110     /**
111      * Receives a <code>ProtocolDataUnit</code> from the socket and appends it
112      * to the end of the receiving queue of this connection.
113      * 
114      * @return Queue with the resulting units
115      * @throws IOException
116      *             if an I/O error occurs.
117      * @throws InternetSCSIException
118      *             if any violation of the iSCSI-Standard emerge.
119      * @throws DigestException
120      *             if a mismatch of the digest exists.
121      * @throws SettingsException
122      */
123     ProtocolDataUnit receiveFromWire() throws DigestException, InternetSCSIException, IOException,
124         SettingsException {
125 
126         ProtocolDataUnit pdu;
127         if (initialPdu) {
128             /*
129              * The connection's ConnectionSettingsNegotiator has not been
130              * initialized, hence getSettings() would throw a
131              * NullPointerException.
132              * 
133              * Initialize PDU with default values, i.e. no digests.
134              */
135             pdu = protocolDataUnitFactory.create(TextKeyword.NONE,// header
136                                                                   // digest
137                 TextKeyword.NONE);// data digest
138         } else {
139             // use negotiated or (now available) default settings
140             final Settings settings = connection.getSettings();
141             pdu = protocolDataUnitFactory.create(settings.getHeaderDigest(), settings.getDataDigest());
142         }
143 
144         try {
145             pdu.read(socketChannel);
146         } catch (ClosedChannelException e) {
147             throw new InternetSCSIException(e);
148         }
149 
150         if (LOGGER.isDebugEnabled())
151             LOGGER.debug("Receiving this PDU:\n" + pdu);
152 
153         // parse sequence counters
154         final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
155         final InitiatorMessageParser parser = (InitiatorMessageParser)bhs.getParser();
156         final int commandSequenceNumber = parser.getCommandSequenceNumber();
157         final int expectedStatusSequenceNumber = parser.getExpectedStatusSequenceNumber();
158 
159         if (LOGGER.isDebugEnabled()) {
160             // sharrajesh
161             // Needed to debug, out of order receiving of StatusSN and ExpStatSN
162             if (bhs.getOpCode() == OperationCode.SCSI_COMMAND) {
163                 final SCSICommandParser scsiParser = (SCSICommandParser)bhs.getParser();
164                 ScsiOperationCode scsiOpCode = ScsiOperationCode.valueOf(scsiParser.getCDB().get(0));
165                 LOGGER.debug("scsiOpCode = " + scsiOpCode);
166                 LOGGER.debug("CDB bytes: \n" + Debug.byteBufferToString(scsiParser.getCDB()));
167             }
168             LOGGER.debug("parser.expectedStatusSequenceNumber: " + expectedStatusSequenceNumber);
169             if (connection == null)
170                 LOGGER.debug("connection: null");
171             else if (connection.getStatusSequenceNumber() == null)
172                 LOGGER.debug("connection.getStatusSequenceNumber: null");
173             else
174                 LOGGER.debug("connection.getStatusSequenceNumber: "
175                     + connection.getStatusSequenceNumber().getValue());
176         }
177 
178         // if this is the first PDU in the leading connection, then
179         // initialize the session's ExpectedCommandSequenceNumber
180         if (initialPdu) {
181             initialPdu = false;
182             // PDU is immediate Login PDU, checked in Target.main(),
183             // ExpCmdSN of this PDU will be used to initialize the
184             // respective session and connection parameters (sequence numbers)
185             // see TargetSession and TargetConnection initialization in
186             // Target.main()
187         } else {
188             // check sequence counters
189             if (session.getMaximumCommandSequenceNumber().lessThan(commandSequenceNumber))
190                 throw new InternetSCSIException("received CmdSN > local MaxCmdSN");
191 
192             // verified, is working with Windows 8 initiator
193             // if (!connection.getStatusSequenceNumber().equals(expectedStatusSequenceNumber)
194             // && expectedStatusSequenceNumber != 0)// required by MS iSCSI
195             // // initiator DATA-OUT
196             // // PDU sequence
197             // throw new InternetSCSIException("received ExpStatusSN != local StatusSN + 1");
198         }
199 
200         // increment CmdSN if not immediate PDU (or Data-Out PDU)
201         if (parser.incrementSequenceNumber())
202             session.getExpectedCommandSequenceNumber().increment();
203 
204         return pdu;
205     }

207     /**
208      * Sends the given <code>ProtocolDataUnit</code> instance over the socket to
209      * the connected iSCSI Target.
210      * 
211      * @param pdu
212      *            The <code>ProtocolDataUnit</code> instances to send.
213      * @throws InternetSCSIException
214      *             if any violation of the iSCSI-Standard emerge.
215      * @throws IOException
216      *             if an I/O error occurs.
217      * @throws InterruptedException
218      *             if another caller interrupted the current caller before or
219      *             while the current caller was waiting for a notification. The
220      *             interrupted status of the current caller is cleared when this
221      *             exception is thrown.
222      */
223 
224     final void sendOverWire(final ProtocolDataUnit pdu) throws InternetSCSIException, IOException,
225         InterruptedException {
226 
227         // set sequence counters
228         final TargetMessageParser parser = (TargetMessageParser)pdu.getBasicHeaderSegment().getParser();
229         parser.setExpectedCommandSequenceNumber(session.getExpectedCommandSequenceNumber().getValue());
230         parser.setMaximumCommandSequenceNumber(session.getMaximumCommandSequenceNumber().getValue());
231         final boolean incrementSequenceNumber = parser.incrementSequenceNumber();
232         if (incrementSequenceNumber)// set StatSN only if field is not reserved
233             parser.setStatusSequenceNumber(connection.getStatusSequenceNumber().getValue());
234 
235         if (LOGGER.isDebugEnabled())
236             LOGGER.debug("Sending this PDU:\n" + pdu);
237 
238         // send pdu
239         pdu.write(socketChannel);
240 
241         // increment StatusSN if this was a Response PDU (with status)
242         // or if special cases apply
243         if (incrementSequenceNumber)
244             connection.getStatusSequenceNumber().increment();
245 
246     }
